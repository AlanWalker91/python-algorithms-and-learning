## 2. 变量不是盒子——变量是标注

```python
# =============================================
# 关键理解：Python 变量是引用，不是存储容器
# =============================================

# ---- 示例1：别名演示 ----
a = [1, 2, 3]     # 创建列表，贴上标签 a
b = a              # b 和 a 指向同一个列表对象（别名！）
a.append(4)
print(b)           # [1, 2, 3, 4]（b 看到了修改！）
# 如果 b 是盒子（存储副本），b 应该是 [1, 2, 3]，但实际不是。

# ---- 示例2：赋值顺序（先创建对象，再绑定变量）----
class Gizmo:
    def __init__(self):
        print(f'Gizmo id: {id(self)}')

x = Gizmo()          # 正常：创建对象，绑定到 x
# y = Gizmo() * 10   # 先创建 Gizmo，然后乘法抛出 TypeError
                     # 所以 y 不存在，但 Gizmo() 已经创建了！

# 结论：赋值语句右边先执行（创建对象），
# 左边的变量才绑定到这个已创建的对象上。

# ---- 变量比喻 ----
# 把变量想象为便利贴（标注），不是盒子：
# - 盒子：值存储在盒子里，赋值是把内容放进盒子
# - 便利贴：对象先存在，变量是贴在对象上的标注
# b = a 是把标注 b 贴到 a 指向的对象上，不是复制对象
```

## 3. 同一性、相等性和别名

```python
# =============================================
# is vs ==：两种不同的比较
# =============================================

# is：比较对象的 ID（标识，即内存地址）
# ==：比较对象的值（调用 __eq__ 方法）

charles = {'name': 'Charles L. Dodgson', 'born': 1832}
lewis = charles         # lewis 是 charles 的别名

print(lewis is charles)     # True（同一个对象）
print(id(charles), id(lewis))  # 相同的 ID

lewis['balance'] = 950
print(charles)   # {'name': 'Charles L. Dodgson', 'born': 1832, 'balance': 950}
# 修改 lewis 等于修改 charles（因为是同一个对象）

# 创建一个具有相同内容但不同标识的对象
alex = {'name': 'Charles L. Dodgson', 'born': 1832, 'balance': 950}

print(alex == charles)      # True（值相等）
print(alex is not charles)  # True（不是同一个对象）

# =============================================
# 何时用 is，何时用 ==
# =============================================

# ✅ is 的正确用法：测试 None 和哨符对象
x = None
print(x is None)       # True（推荐）
print(x is not None)   # False

# 自定义哨符对象（单例）
END_OF_DATA = object()
def traverse(node):
    if node is END_OF_DATA:  # ✅ 用 is 比较哨符
        return

# ⚠️ is 的速度更快（直接比较整数），但不能用于值比较
# a is b 等价于 id(a) == id(b)，不等价于 a == b

# ❌ 常见错误：用 is 比较字符串或整数值
# 虽然某些情况下可行（驻留优化），但这是不安全的
s1 = 'hello'
s2 = 'hello'
print(s1 is s2)    # True（但这是实现细节，不可依赖！）
print(s1 == s2)    # True（正确的比较方式）
```

## 4. 元组的相对不可变性

```python
# =============================================
# 元组不可变是指：引用（item 的 ID）不可变
# 但引用的可变对象，其内容可以改变
# =============================================

t1 = (1, 2, [30, 40])
t2 = (1, 2, [30, 40])

print(t1 == t2)         # True（值相等）
print(id(t1[-1]))       # 某个 ID（列表的地址）

t1[-1].append(99)       # 修改 t1 中的列表（列表是可变的）
print(t1)               # (1, 2, [30, 40, 99])
print(id(t1[-1]))       # 同一个 ID（引用没变，只是列表内容变了）

print(t1 == t2)         # False（值不再相等）

# ---- 实践意义 ----
# 含可变项的元组不可哈希：
t = (1, [2, 3])
try:
    hash(t)             # TypeError: unhashable type: 'list'
except TypeError as e:
    print(e)

# 判断元组是否"值固定"（可哈希）：
def fixed(o):
    try:
        hash(o)
        return True
    except TypeError:
        return False

print(fixed((1, 2, (3, 4))))   # True（全部不可变）
print(fixed((1, 2, [3, 4])))   # False（含列表）
```

## 5. 默认做浅拷贝

```python
import copy

# =============================================
# 浅拷贝：复制外层容器，但内部对象共享引用
# =============================================

l1 = [3, [66, 55, 44], (7, 8, 9)]
l2 = list(l1)         # 浅拷贝（等价于 l1[:]）

print(l1 == l2)       # True（值相等）
print(l1 is l2)       # False（不同对象）

# 操作演示（关键！）
l1.append(100)         # 修改 l1 外层，l2 不受影响
print(l1, '\n', l2)    # l2 没有 100

l1[1].remove(55)       # 修改共享的内部列表！l2 也受影响！
print(l1[1], l2[1])   # 两者都变为 [66, 44]

l2[1] += [33, 22]      # 就地修改 l2[1]（共享列表），l1[1] 也变！
print(l1[1], l2[1])   # 两者都变为 [66, 44, 33, 22]

l2[2] += (10, 11)      # 元组 +=：创建新元组并重新绑定，l1[2] 不变！
print(l1[2], l2[2])   # l1[2]=(7,8,9)，l2[2]=(7,8,9,10,11)

# =============================================
# 深拷贝：递归复制所有对象，不共享内部引用
# =============================================

class Bus:
    """运载乘客的校车。"""
    def __init__(self, passengers=None):
        if passengers is None:
            self.passengers = []
        else:
            self.passengers = list(passengers)   # 防御性复制

    def pick(self, name):
        self.passengers.append(name)

    def drop(self, name):
        self.passengers.remove(name)


bus1 = Bus(['Alice', 'Bill', 'Claire', 'David'])
bus2 = copy.copy(bus1)       # 浅拷贝
bus3 = copy.deepcopy(bus1)   # 深拷贝

bus1.drop('Bill')

# 浅拷贝：bus1 和 bus2 共享同一个 passengers 列表！
print(bus2.passengers)       # ['Alice', 'Claire', 'David']（受影响！）

# 深拷贝：bus3 有独立的 passengers 列表
print(bus3.passengers)       # ['Alice', 'Bill', 'Claire', 'David']（不受影响）

# 深拷贝能处理循环引用！
a = [10, 20]
b = [a, 30]
a.append(b)              # 循环引用：a → b → a
print(a)                 # [10, 20, [[...], 30]]

c = copy.deepcopy(a)     # 不会无限递归！deepcopy 记录已复制的对象
print(c)                 # [10, 20, [[...], 30]]
```

## 6. 函数的参数是引用时（共享传参）

### 6.1 基本行为

```python
# =============================================
# Python 使用"共享传参"（call by sharing）：
# 形参是实参引用的副本（别名）
# 函数可修改可变对象，但不能替换对象本身
# =============================================

def f(a, b):
    a += b
    return a

# 数值（不可变）：函数内修改不影响外部
x, y = 1, 2
f(x, y)
print(x, y)          # 1 2（x 不变）

# 列表（可变）：函数内就地修改会影响外部！
a, b = [1, 2], [3, 4]
f(a, b)
print(a, b)          # [1, 2, 3, 4] [3, 4]（a 被就地修改了！）

# 元组（不可变）：函数内 += 创建新对象，不影响外部
t, u = (10, 20), (30, 40)
f(t, u)
print(t, u)          # (10, 20) (30, 40)（t 不变）
```

### 6.2 ⚠️ 可变默认参数陷阱

```python
# ❌ 危险：可变类型作为默认参数
class HauntedBus:
    """受幽灵乘客折磨的校车（错误示例）。"""
    def __init__(self, passengers=[]):   # ← 所有实例共享同一个列表！
        self.passengers = passengers

    def pick(self, name):
        self.passengers.append(name)

    def drop(self, name):
        self.passengers.remove(name)


bus1 = HauntedBus(['Alice', 'Bill'])
bus1.drop('Alice')
print(bus1.passengers)    # ['Bill']  正常

bus2 = HauntedBus()       # 使用默认空列表
bus2.pick('Carrie')
print(bus2.passengers)    # ['Carrie']  正常

bus3 = HauntedBus()       # 又用了同一个默认列表！
print(bus3.passengers)    # ['Carrie']  ← 幽灵乘客！

# 原因：默认值在函数定义时求解（通常是加载模块时），
# 成为函数对象的属性，所有调用共享同一个默认列表对象！
print(HauntedBus.__init__.__defaults__)  # (['Carrie'],) ← 已被污染！


# ✅ 正确做法：用 None 作为默认值
class SafeBus:
    def __init__(self, passengers=None):     # ← None 作为默认值
        if passengers is None:
            self.passengers = []             # 每次创建新列表
        else:
            self.passengers = list(passengers)  # 防御性复制
```

### 6.3 防御可变参数

```python
# ❌ 危险：直接使用传入的列表，与调用方共享
class TwilightBus:
    def __init__(self, passengers=None):
        if passengers is None:
            self.passengers = []
        else:
            self.passengers = passengers    # ← 直接赋值（别名）！

# 调用方不知道 Bus 内部会修改他们的列表！
team = ['Sue', 'Tina', 'Maya']
bus = TwilightBus(team)
bus.drop('Tina')
print(team)     # ['Sue', 'Maya']  ← team 被悄悄修改了！


# ✅ 正确做法：复制传入的参数
class GoodBus:
    def __init__(self, passengers=None):
        if passengers is None:
            self.passengers = []
        else:
            self.passengers = list(passengers)  # ← 创建副本！

team = ['Sue', 'Tina', 'Maya']
bus = GoodBus(team)
bus.drop('Tina')
print(team)     # ['Sue', 'Tina', 'Maya']  ← team 不受影响 ✅
```

## 7. del 和垃圾回收

```python
# =============================================
# del 删除引用，不删除对象
# 对象引用计数归零时，CPython 才销毁它
# =============================================

a = [1, 2]
b = a             # b 也引用同一个列表

del a             # 删除引用 a（列表还有 b 引用着）
print(b)          # [1, 2]（列表仍然存在）

b = [3]           # b 重新绑定，[1, 2] 引用计数归零 → 被销毁

# ---- CPython 垃圾回收机制 ----
# 主要算法：引用计数
# - 每个对象统计有多少引用指向自己
# - 引用计数归零 → 立即调用 __del__（如果定义了）→ 释放内存
# - 优点：及时回收；缺点：无法处理循环引用

# 分代垃圾回收器（CPython 2.0+）：
# - 处理循环引用中不可达的对象组
# - 在引用计数之外额外运行

# ---- 演示对象生命结束 ----
import weakref

s1 = {1, 2, 3}
s2 = s1           # s2 是别名

def bye():
    print('...对象被销毁了')

ender = weakref.finalize(s1, bye)   # 注册销毁回调
print(ender.alive)    # True

del s1                              # 删除引用 s1（s2 还指向它）
print(ender.alive)    # True（还有 s2 引用）

s2 = 'spam'           # 重新绑定 s2，{1,2,3} 引用计数归零
# 打印：...对象被销毁了
print(ender.alive)    # False

# ⚠️ 注意：__del__ 方法
# - 不该由用户代码直接调用
# - Python 解释器在对象即将销毁前调用
# - 不保证一定被调用（如解释器退出时）
```

## 8. 弱引用

```python
import weakref

# =============================================
# 弱引用：不增加引用计数，不阻止对象被回收
# 用途：缓存（不想因缓存而阻止对象销毁）
# =============================================

class Widget:
    """一个简单的类，演示弱引用。"""
    pass

obj = Widget()                       # 强引用
wref = weakref.ref(obj)              # 弱引用（不增加引用计数）

print(wref)        # <weakref at 0x...; to 'Widget' at 0x...>
print(wref())      # <Widget object>（调用弱引用获取对象）

del obj            # 删除强引用，引用计数变为 0，对象被销毁
print(wref())      # None（对象已被销毁，弱引用返回 None）

# ---- WeakValueDictionary：缓存场景 ----
class Cheese:
    def __init__(self, kind):
        self.kind = kind
    def __repr__(self):
        return f'Cheese({self.kind!r})'

stock = weakref.WeakValueDictionary()   # 值是弱引用
catalog = [Cheese('Red Leicester'), Cheese('Tilsit'), Cheese('Brie')]

for cheese in catalog:
    stock[cheese.kind] = cheese         # 弱引用存入字典

print(sorted(stock.keys()))
# ['Brie', 'Red Leicester', 'Tilsit']

del catalog     # 删除强引用（catalog 是唯一的强引用）
# 所有 Cheese 对象的引用计数归零，被销毁

print(sorted(stock.keys()))
# [] ← 字典自动清空了（弱引用不阻止回收）
```

## 9. Python 对不可变类型施加的把戏

```python
# =============================================
# CPython 对不可变类型的优化：共享对象（驻留）
# 不要依赖这些行为！
# =============================================

# ---- 元组共享 ----
t1 = (1, 2, 3)
t2 = tuple(t1)        # 不创建副本，返回同一对象的引用
print(t2 is t1)       # True（实现优化！）

t3 = t1[:]
print(t3 is t1)       # True（切片也是同一对象）

# ---- 字符串驻留 ----
s1 = 'ABC'
s2 = 'ABC'
print(s2 is s1)       # True（驻留优化：共享字符串字面量）

# 但不是所有字符串都会被驻留：
s3 = 'hello world'    # 含空格的字符串可能不驻留
s4 = 'hello world'
print(s3 is s4)       # 结果不确定，取决于 Python 实现

# ⚠️ 千万不要依赖驻留行为！
# 比较字符串或整数应该用 ==，不用 is
# 驻留只是实现细节，随时可能改变
```

## 10. 常见误区与陷阱

### ❌ 误区一：以为赋值就是复制

```python
# ❌ 错误认知
a = [1, 2, 3]
b = a              # 不是复制！是别名！
b.append(4)
print(a)           # [1, 2, 3, 4]  ← a 也变了！

# ✅ 真正复制需要显式操作
b = a[:]           # 浅拷贝
b = list(a)        # 浅拷贝
import copy
b = copy.deepcopy(a)  # 深拷贝
```

### ❌ 误区二：用 `is` 比较值

```python
# ❌ 错误：用 is 比较整数或字符串值
x = 1000
y = 1000
print(x is y)    # 可能为 False（超出小整数驻留范围）

# ✅ 正确：用 == 比较值
print(x == y)    # True（比较值）

# ✅ is 只用于 None 和哨符对象
print(x is None)  # False（正确用法）
```

### ❌ 误区三：函数修改可变参数（意外副作用）

```python
# ❌ 危险：函数悄悄修改了传入的列表
def process(data):
    data.sort()    # 就地排序！影响调用方！
    return data

my_list = [3, 1, 2]
result = process(my_list)
print(my_list)     # [1, 2, 3]  ← 被修改了！

# ✅ 正确：在函数内创建副本
def safe_process(data):
    sorted_data = sorted(data)   # 返回新列表，不修改原列表
    return sorted_data

my_list = [3, 1, 2]
result = safe_process(my_list)
print(my_list)     # [3, 1, 2]  ← 未被修改 ✅
```

### ❌ 误区四：可变默认参数

```python
# ❌ 危险：可变默认参数被所有调用共享
def add_item(item, items=[]):     # ← 危险！
    items.append(item)
    return items

print(add_item('a'))   # ['a']
print(add_item('b'))   # ['a', 'b']  ← 保留了上次的结果！

# ✅ 正确：用 None 作为默认值
def add_item_safe(item, items=None):
    if items is None:
        items = []
    items.append(item)
    return items

print(add_item_safe('a'))   # ['a']
print(add_item_safe('b'))   # ['b']  ← 独立的列表 ✅
```

### ❌ 误区五：以为 del 删除对象

```python
# ❌ 错误认知：del x 销毁了对象
x = [1, 2, 3]
y = x
del x              # 删除引用 x，对象还在！
print(y)           # [1, 2, 3]  ← 对象通过 y 仍可访问

# del 只删除引用（标注），对象在引用计数归零后才被销毁
```
