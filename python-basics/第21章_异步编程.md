## 2. 核心定义：三种协程与异步对象

### 2.1 原生协程（Native Coroutine）⭐

```python
# 使用 async def 定义，是本章的核心
async def fetch_data(url: str) -> bytes:
    """原生协程示例。

    - 通过 await 把控制权让给事件循环
    - 调用协程函数返回协程对象，不会立即执行
    - 必须由事件循环驱动（asyncio.run / await / create_task）
    """
    async with httpx.AsyncClient() as client:
        resp = await client.get(url)    # await 挂起，让事件循环处理其他任务
        return resp.content


# 调用协程函数不会执行主体！只创建协程对象
coro = fetch_data('https://example.com')   # → 返回 coroutine 对象
result = await coro                         # → 真正执行（只能在async def内）
# 或者
result = asyncio.run(fetch_data('https://example.com'))  # 在普通函数中驱动
```

### 2.2 异步生成器（Async Generator）

```python
import asyncio
from typing import AsyncIterator

async def multi_fetch(urls: list[str]) -> AsyncIterator[bytes]:
    """异步生成器：逐一产出每个URL的响应。

    - 使用 async def + yield 定义
    - 调用返回 async_generator 对象
    - 只能用 async for 或异步推导式驱动
    """
    async with httpx.AsyncClient() as client:
        for url in urls:
            resp = await client.get(url)
            yield resp.content              # yield 使它成为异步生成器


# 使用 async for 驱动异步生成器
async def main():
    urls = ['https://example.com', 'https://python.org']
    async for data in multi_fetch(urls):    # 按完成顺序产出
        print(f'Downloaded {len(data)} bytes')
```

### 2.3 协程类型对比

| | 原生协程 | 异步生成器 |
|--|---------|-----------|
| **定义方式** | `async def`（无 yield）| `async def` + `yield` |
| **调用结果** | `coroutine` 对象 | `async_generator` 对象 |
| **驱动方式** | `await` / `create_task` | `async for` / 异步推导式 |
| **返回值** | `return value`（非None）| 只能 `return`（无值）|
| **可否 await** | ✅ 是 Awaitable | ❌ 不是 Awaitable |
| **用途** | 异步操作的主要形式 | 异步序列生成 / 流式数据 |

## 3. asyncio 示例：探测域名

```python
#!/usr/bin/env python3
"""
blogdom.py：用 asyncio 并发探测多个域名是否可解析。

核心思路：同时发起多个 DNS 查询，哪个先完成就先显示，
总耗时 ≈ 最慢的单次查询时间（而非所有查询时间之和）。
"""
import asyncio
import socket
from keyword import kwlist

MAX_KEYWORD_LEN = 4     # 只检测长度 ≤ 4 的关键字（域名越短越好）


async def probe(domain: str) -> tuple[str, bool]:
    """探测一个域名是否可通过 DNS 解析。

    Returns:
        (domain, True)   ← 域名存在（已被注册）
        (domain, False)  ← 域名不可解析（可能可以注册）
    """
    loop = asyncio.get_running_loop()    # 获取当前运行的事件循环
    try:
        # loop.getaddrinfo 是协程方法，会异步等待 DNS 响应
        # 内部使用 run_in_executor 调用阻塞的 socket.getaddrinfo
        await loop.getaddrinfo(domain, None)
    except socket.gaierror:
        return (domain, False)           # DNS 解析失败 → 域名不存在
    return (domain, True)                # DNS 解析成功 → 域名存在


async def main() -> None:
    # 生成所有长度 ≤ 4 的 Python 关键字对应的 .dev 域名
    names = (kw for kw in kwlist if len(kw) <= MAX_KEYWORD_LEN)
    domains = (f'{name}.dev'.lower() for name in names)

    # 构建协程对象列表（注意：此时协程未执行，只是创建了对象）
    coros = [probe(domain) for domain in domains]

    # asyncio.as_completed：按完成顺序产出协程（先完成先处理）
    for coro in asyncio.as_completed(coros):
        domain, found = await coro      # await 获取已完成协程的结果
        mark = '+' if found else ' '
        print(f'{mark} {domain}')


if __name__ == '__main__':
    asyncio.run(main())     # 启动事件循环，驱动 main() 协程


# Guido 阅读异步代码的技巧：
# 对 await 关键字视而不见，把协程当作普通函数阅读
# async def probe(domain):
#     loop = asyncio.get_running_loop()
#     try:
#         loop.getaddrinfo(domain, None)    # 假装不阻塞
#     except socket.gaierror:
#         return (domain, False)
#     return (domain, True)
```

## 4. 可异步调用对象（Awaitable）

> `await` 关键字处理的对象叫**可异步调用对象**（Awaitable），类似于 `for` 处理**可迭代对象**。

```python
# 两种常见的 Awaitable（终端用户视角）：

# 1. 原生协程对象（调用 async def 函数得到）
async def my_coro():
    return 42

coro_obj = my_coro()        # 协程对象，是 Awaitable
result = await coro_obj     # 驱动协程，得到返回值

# 2. asyncio.Task（由 create_task 创建）
task = asyncio.create_task(my_coro())    # 调度并发执行，立即返回 Task
result = await task                       # 等待 Task 完成

# Task 与直接 await 的区别：
# - await coro()          → 串行执行，当前协程等待它完成
# - create_task(coro())   → 并发调度，立即返回，不等待
#                           当前协程可以继续做其他事

# 实际应用：并发下载
async def download_all(urls):
    tasks = [asyncio.create_task(fetch(url)) for url in urls]  # 创建所有任务
    results = await asyncio.gather(*tasks)                       # 等待全部完成
    return results
```

## 5. 使用 asyncio + HTTPX 下载国旗

### 5.1 设置函数（顶层协调）

```python
import asyncio
from httpx import AsyncClient
from flags import BASE_URL, save_flag, main


def download_many(cc_list: list[str]) -> int:
    """普通函数（非协程），作为程序入口。

    asyncio.run() 启动事件循环，驱动 supervisor 协程。
    """
    return asyncio.run(supervisor(cc_list))


async def supervisor(cc_list: list[str]) -> int:
    """顶层协调协程，并发下载所有国旗。"""
    async with AsyncClient() as client:         # 异步上下文管理器（详见第6节）
        # 为每个国家代码创建一个 download_one 协程对象
        to_do = [download_one(client, cc) for cc in sorted(cc_list)]

        # asyncio.gather：并发等待所有协程，按提交顺序返回结果列表
        # ⭐ 与 asyncio.as_completed 的区别：
        #    gather → 等所有完成后一次性返回（适合需要全部结果的场景）
        #    as_completed → 每完成一个立即产出（适合流式处理/进度条）
        res = await asyncio.gather(*to_do)

    return len(res)


if __name__ == '__main__':
    main(download_many)
```

### 5.2 下载协程（底层执行）

```python
async def download_one(client: AsyncClient, cc: str):
    """下载单个国旗图像的协程。

    必须是协程，因为 get_flag 用了 await。
    "不成功便成仁"：一旦 get_flag 是协程，调用它的函数也必须是协程。
    """
    image = await get_flag(client, cc)      # await 异步HTTP请求
    save_flag(image, f'{cc}.gif')           # ⚠️ 同步文件I/O，会短暂阻塞事件循环
    print(cc, end=' ', flush=True)
    return cc


async def get_flag(client: AsyncClient, cc: str) -> bytes:
    """发起异步HTTP请求，返回图像字节数据。"""
    url = f'{BASE_URL}/{cc}/{cc}.gif'.lower()
    resp = await client.get(url, timeout=6.1, follow_redirects=True)
    # AsyncClient.get() 是协程方法，必须用 await
    return resp.read()
```

### 5.3 "不成功便成仁"原则

```python
# ❌ 错误：普通函数中使用 HTTPX 的异步 API
def get_flag_wrong(cc: str) -> bytes:
    async with AsyncClient() as client:
        resp = await client.get(url)    # SyntaxError：普通函数不能用 await

# ✅ 必须是协程：
async def get_flag_correct(client: AsyncClient, cc: str) -> bytes:
    resp = await client.get(url)        # ← 协程，可以 await
    return resp.read()

# 连锁反应：
# get_flag 是协程 → download_one 必须 await get_flag → download_one 必须是协程
# → supervisor 必须 await download_one → supervisor 必须是协程
# → download_many 必须用 asyncio.run 驱动 supervisor

# 如果某个阻塞函数无法改写为协程（如第三方库的同步API）：
# → 用 asyncio.to_thread() 委托给线程执行（详见第8节）
```

## 6. 异步上下文管理器（async with）

> 普通 `with` 语句的 `__enter__`/`__exit__` 是普通方法，无法 await。  
> `async with` 处理实现了 `__aenter__` 和 `__aexit__` 协程方法的对象。

```python
# 典型用例：数据库事务（asyncpg）

# ❌ 传统写法：需要手动 await
tr = connection.transaction()
await tr.start()
try:
    await connection.execute("INSERT INTO users VALUES (1, 'Alice')")
except:
    await tr.rollback()
    raise
else:
    await tr.commit()

# ✅ 使用 async with：自动处理 __aenter__ 和 __aexit__
async with connection.transaction():
    await connection.execute("INSERT INTO users VALUES (1, 'Alice')")
# 退出时自动 commit（无异常）或 rollback（有异常）


# HTTPX 的 AsyncClient 也是异步上下文管理器：
async with httpx.AsyncClient() as client:
    resp = await client.get('https://example.com')
    # 退出时自动关闭连接池，释放资源
```

### 使用 @asynccontextmanager 创建异步上下文管理器

```python
from contextlib import asynccontextmanager
import asyncio


@asynccontextmanager
async def web_page(url: str):
    """自定义异步上下文管理器：下载网页并更新统计。

    @asynccontextmanager 把异步生成器函数变成异步上下文管理器，
    yield 之前的代码是 __aenter__，yield 之后的代码是 __aexit__。
    """
    loop = asyncio.get_running_loop()

    # __aenter__：在线程中执行阻塞的 download_webpage（防止冻结事件循环）
    data = await loop.run_in_executor(None, download_webpage, url)

    yield data              # 把 data 绑定到 async with ... as data 中的变量

    # __aexit__：在线程中执行阻塞的统计更新
    await loop.run_in_executor(None, update_stats, url)


# 使用：
async def process(url):
    async with web_page(url) as data:
        result = analyze(data)          # 在 with 块内处理数据
```

## 7. 增强版下载器：信号量限流 + 进度条 + 错误处理

### 7.1 信号量（Semaphore）：限制并发数量

```python
# asyncio.Semaphore：限制同时活跃的协程数量
# 内部有一个计数器，acquire() 减1，release() 加1，为0时阻塞

semaphore = asyncio.Semaphore(5)   # 最多5个协程同时持有

async def download_with_limit(client, url):
    async with semaphore:           # acquire() 时计数器减1
        data = await fetch(client, url)
    # 退出 with 块时自动 release()，计数器加1
    return data
```

### 7.2 完整的带错误处理的下载协程

```python
import asyncio
from collections import Counter
from http import HTTPStatus
import httpx
import tqdm
from flags2_common import main, DownloadStatus, save_flag

DEFAULT_CONCUR_REQ = 5
MAX_CONCUR_REQ = 1000


async def get_flag(client: httpx.AsyncClient,
                   base_url: str,
                   cc: str) -> bytes:
    """异步下载国旗图像。"""
    url = f'{base_url}/{cc}/{cc}.gif'.lower()
    resp = await client.get(url, timeout=3.1, follow_redirects=True)
    resp.raise_for_status()         # 非2xx状态码抛出 HTTPStatusError
    return resp.content


async def download_one(client: httpx.AsyncClient,
                       cc: str,
                       base_url: str,
                       semaphore: asyncio.Semaphore,
                       verbose: bool) -> DownloadStatus:
    """下载单个国旗，使用信号量限流。"""
    try:
        async with semaphore:               # 限制并发：最多 concur_req 个协程同时执行
            image = await get_flag(client, base_url, cc)
    except httpx.HTTPStatusError as exc:
        res = exc.response
        if res.status_code == HTTPStatus.NOT_FOUND:
            status = DownloadStatus.NOT_FOUND
            msg = f'not found: {res.url}'
        else:
            raise                           # 其他HTTP错误重新抛出

    else:
        # ⭐ 文件I/O 委托给线程，防止阻塞事件循环
        await asyncio.to_thread(save_flag, image, f'{cc}.gif')
        status = DownloadStatus.OK
        msg = 'OK'

    if verbose and msg:
        print(cc, msg)
    return status
```

### 7.3 完整的顶层 supervisor 协程

```python
async def supervisor(cc_list: list[str],
                     base_url: str,
                     verbose: bool,
                     concur_req: int) -> Counter[DownloadStatus]:
    """顶层协调协程。

    使用 asyncio.as_completed 按完成顺序处理结果，配合 tqdm 显示进度条。
    """
    counter: Counter[DownloadStatus] = Counter()
    semaphore = asyncio.Semaphore(concur_req)   # 创建信号量，限制并发数

    async with httpx.AsyncClient() as client:
        # 创建所有下载协程对象（此时未执行）
        to_do = [
            download_one(client, cc, base_url, semaphore, verbose)
            for cc in sorted(cc_list)
        ]

        # as_completed：产出已完成的协程（按完成顺序）
        to_do_iter = asyncio.as_completed(to_do)

        # tqdm 包装迭代器，显示进度条
        if not verbose:
            to_do_iter = tqdm.tqdm(to_do_iter, total=len(cc_list))

        error: httpx.HTTPError | None = None

        for coro in to_do_iter:
            try:
                status = await coro         # 获取已完成协程的结果（不阻塞）
            except httpx.HTTPStatusError as exc:
                error_msg = 'HTTP error {resp.status_code} {resp.reason_phrase}'
                error_msg = error_msg.format(resp=exc.response)
                error = exc
            except httpx.RequestError as exc:
                error_msg = f'{exc} {type(exc)}'.strip()
                error = exc
            except KeyboardInterrupt:
                break                       # Ctrl+C 优雅退出
            else:
                error_msg = ''

            if error:
                status = DownloadStatus.ERROR
                if verbose:
                    url = str(error.request.url)
                    cc = url.split('/')[-1].split('.')[0].upper()
                    print(f'{cc} error: {error_msg}')

            counter[status] += 1

    return counter


def download_many(cc_list, base_url, verbose, concur_req):
    """普通函数入口，驱动事件循环。"""
    coro = supervisor(cc_list, base_url, verbose, concur_req)
    counts = asyncio.run(coro)
    return counts


if __name__ == '__main__':
    main(download_many, DEFAULT_CONCUR_REQ, MAX_CONCUR_REQ)
```

## 8. 把任务委托给执行器（run_in_executor）

> **场景**：协程中遇到无法改写的阻塞调用（文件I/O、第三方同步库等），需要委托给线程执行，防止冻结事件循环。

### 8.1 asyncio.to_thread（Python 3.9+，推荐）

```python
import asyncio


async def download_one(client, cc):
    image = await get_flag(client, cc)

    # ✅ 把阻塞的文件I/O 委托给默认线程池执行
    # asyncio.to_thread 内部调用 loop.run_in_executor(None, fn, *args)
    await asyncio.to_thread(save_flag, image, f'{cc}.gif')

    return cc


# asyncio.to_thread 的优点：
# - 语法简洁
# - 支持关键字参数
# - 返回协程，可以直接 await
```

### 8.2 loop.run_in_executor（兼容 Python 3.7/3.8）

```python
async def download_one_compat(client, cc):
    image = await get_flag(client, cc)

    # 获取当前事件循环
    loop = asyncio.get_running_loop()

    # run_in_executor(executor, func, *args)
    # executor=None → 使用默认的 ThreadPoolExecutor
    await loop.run_in_executor(None, save_flag, image, f'{cc}.gif')
    # ⚠️ 不支持关键字参数，如果需要，用 functools.partial
    import functools
    await loop.run_in_executor(
        None,
        functools.partial(save_flag, image, filename=f'{cc}.gif')
    )

    return cc


# 使用 ProcessPoolExecutor 处理 CPU 密集型任务
async def cpu_intensive_task(data):
    loop = asyncio.get_running_loop()
    with concurrent.futures.ProcessPoolExecutor() as pool:
        result = await loop.run_in_executor(pool, heavy_compute, data)
    return result
```

### 8.3 委托执行器的适用场景

```
文件 I/O（读写磁盘）          → asyncio.to_thread()
使用 requests 等同步HTTP库    → asyncio.to_thread()
调用第三方同步数据库驱动       → asyncio.to_thread()
CPU 密集型计算（纯Python）     → loop.run_in_executor(ProcessPoolExecutor)
C 扩展释放了 GIL 的计算        → loop.run_in_executor(ThreadPoolExecutor)

⚠️ run_in_executor 的注意事项（Caleb Hattingh 警告）：
使用执行器的协程给人一种可以撤销的假象。
底层线程没有撤销机制，asyncio.run 会等待执行器完全关闭才返回。
如果执行器中有长时间运行的任务，可能导致程序无法正常退出。
```

### 9.2 asyncio TCP 服务器

```python
"""
tcp_mojifinder.py：使用纯 asyncio 实现 TCP Unicode 字符搜索服务。
运行方式：python3 tcp_mojifinder.py
测试方式：telnet localhost 2323
"""
import asyncio
import functools
import sys
from charindex import InvertedIndex, format_results

CRLF = b'\r\n'
PROMPT = b'?> '


async def finder(index: InvertedIndex,
                 reader: asyncio.StreamReader,
                 writer: asyncio.StreamWriter) -> None:
    """处理单个客户端连接的协程。

    每个客户端连接都会创建一个 finder 协程实例，
    多个实例并发运行，服务于多个客户端。
    """
    client = writer.get_extra_info('peername')  # 获取客户端 IP:port

    while True:                                 # 处理一个对话会话
        writer.write(PROMPT)                    # write() 是普通方法，写入缓冲区
        await writer.drain()                    # drain() 是协程，刷新缓冲区（执行实际I/O）

        data = await reader.readline()          # 异步等待客户端输入一行

        if not data:                            # 客户端关闭连接
            break

        try:
            query = data.decode().strip()
        except UnicodeDecodeError:
            query = '\x00'                      # 处理控制字符（如 Ctrl-C）

        print(f'From {client}: {query!r}')

        if query:
            if ord(query[:1]) < 32:            # 收到控制字符，退出循环
                break
            results = await search(query, index, writer)
            print(f'To {client}: {results} results.')

    writer.close()
    await writer.wait_closed()
    print(f'Close {client}.')


async def search(query: str,
                 index: InvertedIndex,
                 writer: asyncio.StreamWriter) -> int:
    """执行搜索，把结果写入客户端。"""
    chars = index.search(query)
    lines = (line.encode() + CRLF for line in format_results(chars))
    writer.writelines(lines)                   # writelines() 是普通方法
    await writer.drain()                       # 必须 await drain()！
    status_line = f'{"─" * 66} {len(chars)} found'
    writer.write(status_line.encode() + CRLF)
    await writer.drain()
    return len(chars)


async def supervisor(index: InvertedIndex, host: str, port: int) -> None:
    """启动服务器并保持运行。"""
    # asyncio.start_server：创建并启动 TCP 服务器
    # 每有新客户端连接，自动调用 finder 协程
    server = await asyncio.start_server(
        functools.partial(finder, index),  # 绑定 index 参数（functools.partial 的典型用法）
        host, port
    )

    addr = server.sockets[0].getsockname()
    print(f'Serving on {addr}. Hit CTRL-C to stop.')

    await server.serve_forever()           # 无限等待，保持服务器运行


def main(host: str = '127.0.0.1', port_arg: str = '2323'):
    port = int(port_arg)
    print('Building index.')
    index = InvertedIndex()                # 构建倒排索引
    try:
        asyncio.run(supervisor(index, host, port))
    except KeyboardInterrupt:
        print('\nServer shut down.')


if __name__ == '__main__':
    main(*sys.argv[1:])
```

## 10. 异步迭代与异步生成器

### 10.1 异步迭代器协议

```python
# 同步迭代器协议（回顾）：
# __iter__() → 返回迭代器
# __next__() → 返回下一个值（普通方法）

# 异步迭代器协议：
# __aiter__() → 返回异步迭代器（普通方法，不是协程）
# __anext__() → 返回下一个值（协程方法，用 await 驱动）

class AsyncRange:
    """自定义异步迭代器（用类实现）。"""
    def __init__(self, n):
        self.n = n
        self.i = 0

    def __aiter__(self):        # 普通方法（不是 async def）
        return self

    async def __anext__(self):  # 协程方法
        if self.i >= self.n:
            raise StopAsyncIteration    # 迭代结束时抛出（类似 StopIteration）
        await asyncio.sleep(0)          # 模拟异步操作
        val = self.i
        self.i += 1
        return val

# 使用
async def main():
    async for i in AsyncRange(5):
        print(i)
```

### 10.2 异步生成器（更简洁的写法）⭐

```python
import asyncio
import socket
from collections.abc import Iterable, AsyncIterator
from typing import NamedTuple


class Result(NamedTuple):
    domain: str
    found: bool


async def probe(domain: str, loop=None) -> Result:
    """探测单个域名。"""
    if loop is None:
        loop = asyncio.get_running_loop()
    try:
        await loop.getaddrinfo(domain, None)
    except socket.gaierror:
        return Result(domain, False)
    return Result(domain, True)


async def multi_probe(domains: Iterable[str]) -> AsyncIterator[Result]:
    """异步生成器：并发探测多个域名，按完成顺序产出结果。

    - async def + yield → 这是异步生成器
    - 调用返回 async_generator 对象
    - 必须用 async for 驱动
    """
    loop = asyncio.get_running_loop()
    coros = [probe(domain, loop) for domain in domains]

    # asyncio.as_completed 是普通生成器（不是异步生成器）
    for coro in asyncio.as_completed(coros):    # 普通 for（不是 async for）
        result = await coro                     # 等待完成的协程
        yield result                            # ← yield 使此函数成为异步生成器


# 使用异步生成器：
async def main():
    domains = ['python.org', 'rust-lang.org', 'golang.org', 'no-lang.invalid']
    async for result in multi_probe(domains):   # async for 驱动异步生成器
        mark = '+' if result.found else ' '
        print(f'{mark} {result.domain}')
```

### 10.3 异步生成器表达式与异步推导式

```python
# 在 async def 函数内或异步控制台中可以使用：

async def find_available_domains(domains):
    # 异步生成器表达式（async for）
    available = (
        name async for name, found in multi_probe(domains)
        if not found                    # 未找到 = 可能可以注册
    )
    async for domain in available:
        print(f'Available: {domain}')

    # 异步列表推导式（await）
    results = [await probe(d) for d in domains]

    # 异步字典推导式
    status = {name: found async for name, found in multi_probe(domains)}

    # 异步集合推导式（结合 await）
    found = {name for name in domains if (await probe(name)).found}
```

## 11. asyncio 之外：Curio 与结构化并发

> Python 的 `async/await` 不限于 asyncio，任何异步框架都可以驱动原生协程。

```python
"""
使用 Curio 重写 blogdom.py：展示 async/await 的框架无关性。
Curio 由 David Beazley 开发，API 更简洁，引入了"结构化并发"。
"""
from curio import run, TaskGroup
import curio.socket as socket
from keyword import kwlist

MAX_KEYWORD_LEN = 4


async def probe(domain: str) -> tuple[str, bool]:
    """与 asyncio 版几乎相同，但使用 curio.socket。"""
    try:
        # curio.socket.getaddrinfo 是顶层协程函数（不需要 loop 对象）
        await socket.getaddrinfo(domain, None)
    except socket.gaierror:
        return (domain, False)
    return (domain, True)


async def main() -> None:
    names = (kw for kw in kwlist if len(kw) <= MAX_KEYWORD_LEN)
    domains = (f'{name}.dev'.lower() for name in names)

    # TaskGroup：结构化并发的核心
    # 确保所有派生的任务都完成（或被撤销）后才退出 with 块
    async with TaskGroup() as group:
        for domain in domains:
            await group.spawn(probe, domain)    # 类似 asyncio.create_task

    # 使用 async for 迭代 TaskGroup，按完成顺序处理结果
        async for task in group:
            domain, found = task.result
            mark = '+' if found else ' '
            print(f'{mark} {domain}')


if __name__ == '__main__':
    run(main())     # Curio 的事件循环入口（类似 asyncio.run）
```

**Curio vs asyncio 的关键区别**：

| | asyncio | Curio |
|--|---------|-------|
| `as_completed` | 函数 `asyncio.as_completed()` | `async for task in TaskGroup()` |
| `gather` | `asyncio.gather(*coros)` | `TaskGroup(wait=all)` |
| 获取所有结果 | `await asyncio.gather(...)` | `group.results` |
| 结构化并发 | 较弱（3.11 PEP 654 改进）| 原生支持（TaskGroup）|

## 13. 异步编程的陷阱与原则

### 13.1 阻塞型调用的代价

> I/O 延迟换算（Ryan Dahl 的分析）：

| 设备 | CPU 循环数 | 换算成"人类"时间 |
|------|-----------|----------------|
| L1 缓存 | 3 | 3 秒 |
| L2 缓存 | 14 | 14 秒 |
| RAM | 250 | 250 秒 |
| 磁盘 | 41,000,000 | 1.3 年 |
| 网络 | 240,000,000 | **7.6 年** |

**结论**：在协程中等待网络响应期间，如果不把控制权交还事件循环（即使用 `await`），那就是在"浪费 7.6 年"让事件循环干等着。

### 13.2 "I/O 密集型系统"是幻觉

```
❌ 错误认知："我的系统是 I/O 密集型的，asyncio 完全适合"

✅ 正确认知：
  - 系统中大多数函数是 I/O 密集型的（等待网络/磁盘）
  - 但任何非平凡系统都有一些 CPU 密集型函数
  - 即使是"简单"的 HTTP 服务器，HTTP 首部解析也是 CPU 密集型的
  - 一个 CPU 密集型函数就能拖慢整个事件循环

真实案例（Yury Selivanov 的 uvloop 基准测试）：
  - 简单回显服务器被认为是"I/O 密集型"
  - 但 HTTP 首部解析（Python实现的 aiohttp）是 CPU 密集型
  - 解析器拖慢了事件循环，导致整体性能大打折扣
  - 解决方案：用 C 实现的 httptools 替换 Python 实现的解析器
```

### 13.3 绕开 CPU 密集型陷阱的策略

```python
# 策略1：委托给线程池（适合 I/O 阻塞 + 少量CPU）
await asyncio.to_thread(blocking_function, arg)

# 策略2：委托给进程池（适合 CPU 密集型）
loop = asyncio.get_running_loop()
with concurrent.futures.ProcessPoolExecutor() as pool:
    result = await loop.run_in_executor(pool, cpu_heavy_function, data)

# 策略3：委托给分布式任务队列（适合生产环境）
# Celery.delay() 把任务发送给 Worker 进程，立即返回
celery_task.delay(large_data)

# 策略4：重写 CPU 密集型函数（终极方案）
# 用 Cython / C / Rust 实现，并释放 GIL

# 策略5：asyncio.sleep(0) 定期让出控制权（权宜之计）
async def cpu_heavy_task(n):
    for i in range(n):
        do_some_work(i)
        if i % 10_000 == 0:
            await asyncio.sleep(0)   # 让出控制权，但会降低性能
```

## 14. 常见误区速查

### ❌ 误区一：在协程中使用同步 sleep

```python
# ❌ 错误：冻结整个事件循环
async def slow():
    time.sleep(3)           # 阻塞事件循环3秒！

# ✅ 正确：让出控制权
async def slow():
    await asyncio.sleep(3)  # 挂起协程，事件循环继续运行
```

### ❌ 误区二：认为 asyncio 可以处理 CPU 密集型任务

```python
# ❌ 错误：CPU 密集型任务直接在协程中运行
async def compute():
    result = sum(range(10_000_000))  # 占用 CPU，冻结事件循环
    return result

# ✅ 正确：委托给进程池
async def compute():
    loop = asyncio.get_running_loop()
    result = await loop.run_in_executor(
        ProcessPoolExecutor(),
        sum, range(10_000_000)
    )
    return result
```

### ❌ 误区三：混用同步和异步 API

```python
# ❌ 错误：在协程中使用同步 HTTP 库
async def fetch(url):
    resp = requests.get(url)    # requests 是同步的，会阻塞事件循环
    return resp.content

# ✅ 正确：使用异步 HTTP 库
async def fetch(url):
    async with httpx.AsyncClient() as client:
        resp = await client.get(url)  # HTTPX 异步API
    return resp.content

# 或者把同步调用委托给线程
async def fetch_sync(url):
    return await asyncio.to_thread(requests.get, url)
```

### ❌ 误区四：忘记 await

```python
# ❌ 错误：忘记 await，协程对象未被驱动
async def main():
    asyncio.sleep(1)       # 没有 await！只是创建了协程对象，什么都没发生
    fetch_data('url')      # 同样的错误！

# ✅ 正确
async def main():
    await asyncio.sleep(1)
    result = await fetch_data('url')
```

### ❌ 误区五：不知道 StreamWriter.write 和 drain 的区别

```python
# asyncio Streams API 中的常见混淆：
# write() → 普通方法，写入缓冲区（不执行实际I/O，不能 await）
# drain() → 协程方法，刷新缓冲区（执行实际I/O，必须 await）

async def send_data(writer, data):
    writer.write(data)          # ← 普通方法，不能 await
    await writer.drain()        # ← 协程方法，必须 await！

    # writelines() 也是普通方法
    writer.writelines([b'line1\r\n', b'line2\r\n'])
    await writer.drain()        # 仍然需要 await drain()
```

### ❌ 误区六：以为 asyncio.gather 和 as_completed 相同

```python
# asyncio.gather：等待所有协程完成，按提交顺序返回结果
results = await asyncio.gather(*coros)   # 返回 [result1, result2, ...]

# asyncio.as_completed：按完成顺序产出协程（先完成先处理）
for coro in asyncio.as_completed(coros):
    result = await coro      # 立即可用，不阻塞

# 选择依据：
# - 需要全部结果后统一处理 → gather
# - 需要流式处理/显示进度条 → as_completed
```
