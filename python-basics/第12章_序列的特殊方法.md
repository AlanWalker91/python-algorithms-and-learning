## 2. 协议和鸭子类型

```python
# =============================================
# 协议（Protocol）：非正式的接口，只在文档中定义
# 鸭子类型：对象有类型，变量没有；只看行为，不看声明
# =============================================

import collections

# FrenchDeck 没有继承任何序列基类
# 只实现了 __len__ 和 __getitem__，就满足了序列协议
Card = collections.namedtuple('Card', ['rank', 'suit'])

class FrenchDeck:
    ranks = [str(n) for n in range(2, 11)] + list('JQKA')
    suits = 'spades diamonds clubs hearts'.split()

    def __init__(self):
        self._cards = [Card(rank, suit) for suit in self.suits
                       for rank in self.ranks]

    def __len__(self):     # 序列协议要求1
        return len(self._cards)

    def __getitem__(self, position):  # 序列协议要求2
        return self._cards[position]


deck = FrenchDeck()
print(len(deck))           # 52（通过 __len__）
print(deck[0])             # Card(rank='2', suit='spades')
print(deck[-1])            # 最后一张
print(deck[:3])            # 前3张（切片支持！）

from random import choice
print(choice(deck))        # 随机一张（序列协议带来的免费功能）

# in 运算符（没有 __contains__，退而顺序扫描）
print(Card('Q', 'hearts') in deck)  # True

# 协议不是全有全无：只需实现协议的一部分
# 例如：只有 __getitem__ 也能支持迭代（不需要 __iter__）
```

## 3. Vector 第1版：基本序列协议

```python
from array import array
import reprlib
import math


class Vector:
    """多维向量类第1版：基本序列协议。"""

    typecode = 'd'   # 存储为双精度浮点

    def __init__(self, components):
        # 内部用 array 存储分量（紧凑、高效）
        self._components = array(self.typecode, components)

    def __iter__(self):
        return iter(self._components)  # 把迭代委托给数组

    def __repr__(self):
        # reprlib.repr 生成有长度限制的表示形式
        components = reprlib.repr(self._components)
        # 从 "array('d', [1.0, 2.0, ...])" 中提取 "[1.0, 2.0, ...]"
        components = components[components.find('['):-1]
        return f'Vector({components})'

    def __str__(self):
        return str(tuple(self))

    def __bytes__(self):
        return (bytes([ord(self.typecode)]) +
                bytes(self._components))

    def __eq__(self, other):
        return tuple(self) == tuple(other)

    def __abs__(self):
        return math.hypot(*self)   # Python 3.8+：math.hypot 接受 N 维

    def __bool__(self):
        return bool(abs(self))

    @classmethod
    def frombytes(cls, octets):
        typecode = chr(octets[0])
        memv = memoryview(octets[1:]).cast(typecode)
        return cls(memv)


# 使用示例
v = Vector([3.1, 4.2])
print(repr(v))             # Vector([3.1, 4.2])

v = Vector(range(10))
print(repr(v))             # Vector([0.0, 1.0, 2.0, 3.0, 4.0, ...])（截断！）
print(abs(Vector([3, 4]))) # 5.0
```

## 4. Vector 第2版：切片支持

```python
import operator

class Vector:
    # （前略，同第1版）

    def __len__(self):
        return len(self._components)

    def __getitem__(self, key):
        """支持整数索引和切片。

        切片时返回 Vector 实例（而不是 array），
        以便切片后仍能调用 Vector 的所有方法。
        """
        if isinstance(key, slice):
            cls = type(self)
            # 把切片操作委托给内部数组，再用切片结果构建新 Vector
            return cls(self._components[key])

        # 使用 operator.index 获取整数索引（比 int() 更严格）
        # operator.index(3.14) 会抛出 TypeError（float 不是索引）
        # int(3.14) 会返回 3（静默转换）
        index = operator.index(key)
        return self._components[index]


# 使用示例
v = Vector(range(7))
print(v[1])         # 1.0（整数索引，返回 float）
print(v[1:4])       # Vector([1.0, 2.0, 3.0])（切片，返回 Vector！）
print(v[-1:])       # Vector([6.0])（单元素切片也返回 Vector）

try:
    v[1, 2]         # TypeError：tuple 不是整数索引
except TypeError as e:
    print(e)
```

## 5. 切片原理（`slice` 对象）

```python
# =============================================
# s[a:b:c] 在内部被转换为 slice(a, b, c)
# 然后传给 __getitem__ 方法
# =============================================

class MySeq:
    def __getitem__(self, index):
        return index   # 直接返回，观察传入的值

s = MySeq()
print(s[1])        # 1（整数）
print(s[1:4])      # slice(1, 4, None)（切片对象！）
print(s[1:4:2])    # slice(1, 4, 2)
print(s[1:4:2, 9]) # (slice(1, 4, 2), 9)（逗号 → 元组）
print(s[1:4, 7:9]) # (slice(1, 4, None), slice(7, 9, None))

# slice 对象的属性：start、stop、step
sl = slice(1, 10, 2)
print(sl.start, sl.stop, sl.step)  # 1 10 2

# slice.indices(len)：处理边界问题
# 返回 (start, stop, step)，调整超出边界的索引
print(slice(None, 10, 2).indices(5))  # (0, 5, 2)
print(slice(-3, None, None).indices(5))  # (2, 5, 1)

# ---- 命名切片：提高可读性（消除"魔法数字"）----
invoice = """
0.....6.................................40........52...55........
1909 Pimoroni PiBrella                  $17.50      3      $52.50
1489 6mm Tactile Switch x20             $4.95       2       $9.90
"""

SKU         = slice(0, 6)
DESCRIPTION = slice(6, 40)
UNIT_PRICE  = slice(40, 52)
QUANTITY    = slice(52, 55)
ITEM_TOTAL  = slice(55, None)

for line in invoice.split('\n')[2:]:
    if line:
        print(line[UNIT_PRICE], line[DESCRIPTION])
```

## 6. Vector 第3版：动态属性（`__getattr__`）

```python
class Vector:
    # （前略）

    # 声明 __match_args__：位置模式匹配按此顺序匹配
    __match_args__ = ('x', 'y', 'z', 't')

    def __getattr__(self, name):
        """当常规属性查找失败时调用（兜底机制）。

        通过 x, y, z, t 访问前 4 个分量。
        注意：只在属性查找失败时才调用（有实例属性时不调用）！
        """
        cls = type(self)
        try:
            # 查找 name 在 __match_args__ 中的位置
            pos = cls.__match_args__.index(name)
        except ValueError:
            pos = -1   # 不在列表中

        if 0 <= pos < len(self._components):
            return self._components[pos]

        msg = f'{cls.__name__!r} object has no attribute {name!r}'
        raise AttributeError(msg)

    def __setattr__(self, name, value):
        """防止通过 v.x = 10 这样的赋值造成前后矛盾。

        __getattr__ 的兜底行为：只有属性不存在时才调用。
        如果我们允许 v.x = 10（创建实例属性 x），
        那么下次访问 v.x 就不会调用 __getattr__ 了
        （直接返回实例属性 10），造成与 _components 不一致。
        因此要禁止为单字母小写属性赋值。
        """
        cls = type(self)
        if len(name) == 1:
            if name in cls.__match_args__:
                error = 'readonly attribute {attr_name!r}'
            elif name.islower():
                error = "can't set attributes 'a' to 'z' in {cls_name!r}"
            else:
                error = ''
            if error:
                msg = error.format(cls_name=cls.__name__, attr_name=name)
                raise AttributeError(msg)
        super().__setattr__(name, value)   # 默认行为


# 使用示例
v = Vector(range(5))
print(v.x)    # 0.0（第一个分量）
print(v.y)    # 1.0（第二个分量）
print(v.z)    # 2.0
print(v.t)    # 3.0

try:
    v.x = 10  # AttributeError：只读属性
except AttributeError as e:
    print(e)

try:
    v.a = 10  # AttributeError：禁止单字母小写属性
except AttributeError as e:
    print(e)

v.Z = 99     # ✅ 大写属性可以（不在保护范围内）
```

## 7. Vector 第4版：哈希与高效等值测试

```python
import functools
import operator

class Vector:
    # （前略）

    def __eq__(self, other):
        """高效等值测试（避免为大型向量构建完整元组）。

        先比较长度，再逐个比较分量（短路：遇到不等立即停止）。
        """
        if len(self) != len(other):
            return False
        # zip 并行迭代两个向量，all 短路
        return all(a == b for a, b in zip(self, other))
        # 等价于（更简洁）：
        # return len(self) == len(other) and all(a == b for a, b in zip(self, other))

    def __hash__(self):
        """映射-归约（map-reduce）计算哈希：
        1. 映射：计算各分量的哈希值
        2. 归约：用 XOR 聚合所有哈希值
        """
        hashes = (hash(x) for x in self._components)   # 惰性映射
        return functools.reduce(operator.xor, hashes, 0)
        # reduce(xor, [h0, h1, h2, h3], 0) = ((0 ^ h0) ^ h1) ^ h2 ^ h3

        # 等价写法：
        # hashes = map(hash, self._components)  # map 也是惰性的
        # return functools.reduce(operator.xor, hashes)


# 哈希演示
v1 = Vector([3, 4, 5])
v2 = Vector([3, 4, 5])
print(v1 == v2)          # True
print(hash(v1))          # 某个整数
print(hash(v1) == hash(v2))   # True

# 可用于字典键和集合
d = {v1: '三维点'}
print(d[v2])             # '三维点'

# ---- reduce 原理 ----
# reduce(fn, [a, b, c, d]) = fn(fn(fn(a, b), c), d)
# reduce(xor, [h0, h1, h2], 0) = xor(xor(xor(0, h0), h1), h2)
#                                = 0 ^ h0 ^ h1 ^ h2
import functools, operator
print(functools.reduce(operator.xor, range(6)))  # 1（0^1^2^3^4^5）
```

## 8. Vector 第5版：格式化（超球坐标）

```python
import math

class Vector:
    # （前略）

    def angle(self, n):
        """计算第 n 个球坐标角度（弧度）。"""
        r = math.hypot(*self[n:])
        a = math.atan2(r, self[n-1])
        if (n == len(self) - 1) and (self[-1] < 0):
            return math.pi * 2 - a
        else:
            return a

    def angles(self):
        """返回所有角坐标的可迭代对象。"""
        return (self.angle(n) for n in range(1, len(self)))

    def __format__(self, fmt_spec=''):
        """格式化说明符：
        - 以 'h' 结尾：超球坐标 <r, φ₁, φ₂, ...>
        - 否则：笛卡儿坐标 (x₁, x₂, ...)
        """
        if fmt_spec.endswith('h'):       # 'h' → 超球坐标
            fmt_spec = fmt_spec[:-1]
            coords = itertools.chain([abs(self)], self.angles())
            outer_fmt = '<{}>'
        else:
            coords = self
            outer_fmt = '({})'
        components = (format(c, fmt_spec) for c in coords)
        return outer_fmt.format(', '.join(components))


# 使用示例
v = Vector([3, 4, 5])
print(format(v))             # (3.0, 4.0, 5.0)
print(format(v, '.2f'))      # (3.00, 4.00, 5.00)
print(format(v, '.3eh'))     # <7.071e+00, 9.553e-01, 7.854e-01>（超球坐标）
```

## 9. 常见误区与陷阱

### ❌ 误区一：`__getattr__` 和 `__getattribute__` 混淆

```python
# __getattr__：只在常规属性查找失败后调用（兜底）
# __getattribute__：每次属性访问都调用（非常强大，非常危险）

class Safe:
    def __getattr__(self, name):   # ← 只在找不到属性时调用
        return f'没有属性: {name}'

class Dangerous:
    def __getattribute__(self, name):  # ← 每次属性访问都调用！
        print(f'访问: {name}')
        return super().__getattribute__(name)  # 必须调用 super，否则无限递归

s = Safe()
s.x = 1
print(s.x)       # 1（直接从实例字典读取，不调用 __getattr__）
print(s.y)       # '没有属性: y'（找不到 y，调用 __getattr__）
```

### ❌ 误区二：实现 `__getattr__` 而不实现 `__setattr__`

```python
# ❌ 危险：允许赋值会造成前后矛盾
class Bad:
    def __init__(self, components):
        self._components = list(components)

    def __getattr__(self, name):
        shortcuts = {'x': 0, 'y': 1}
        if name in shortcuts:
            return self._components[shortcuts[name]]
        raise AttributeError(name)

b = Bad([1, 2])
print(b.x)    # 1（调用 __getattr__）
b.x = 99      # 创建了实例属性 x=99！
print(b.x)    # 99（不再调用 __getattr__，直接返回实例属性）
print(b._components[0])  # 1（_components 未变，前后矛盾！）

# ✅ 正确：配套实现 __setattr__ 防止赋值
```

### ❌ 误区三：切片返回错误类型

```python
# ❌ 切片 Vector 返回 array，失去了 Vector 的所有方法
class BadVector:
    def __getitem__(self, key):
        return self._components[key]   # 直接委托，返回 array 类型

# ✅ 切片时应该返回同类型实例
class GoodVector:
    def __getitem__(self, key):
        if isinstance(key, slice):
            cls = type(self)
            return cls(self._components[key])  # 包装成 Vector
        return self._components[operator.index(key)]
```

### ❌ 误区四：`reduce` 的初始值缺失

```python
import functools, operator

# ❌ 危险：没有初始值，空序列会抛出 TypeError
try:
    functools.reduce(operator.xor, [])
except TypeError as e:
    print(e)   # reduce() of empty iterable with no initial value

# ✅ 正确：提供初始值（XOR 的幺元是 0）
print(functools.reduce(operator.xor, [], 0))   # 0

# 对于 + 和 |，幺元是 0；对于 * 和 &，幺元是 1
```
