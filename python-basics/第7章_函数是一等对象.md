# 第7章：函数是一等对象

> **核心思想**：在 Python 中，函数是一等对象——可以在运行时创建、赋值给变量、作为参数传入或作为返回值。这一特性让函数式编程风格（高阶函数、闭包、装饰器）成为自然选择。

## 2. 把函数视为对象

```python
# =============================================
# 函数是 function 类的实例，是一等对象
# =============================================

def factorial(n):
    """返回 n!（n 的阶乘）。"""
    return 1 if n < 2 else n * factorial(n - 1)

# 1. 函数有 __doc__ 属性（文档字符串）
print(factorial.__doc__)    # '返回 n!（n 的阶乘）。'

# 2. 函数是 function 类的实例
print(type(factorial))      # <class 'function'>

# 3. 赋值给变量（另一个名称）
fact = factorial
print(fact(5))              # 120

# 4. 作为参数传给另一个函数
print(list(map(factorial, range(6))))  # [1, 1, 2, 6, 24, 120]

# 5. 函数的常用属性
print(factorial.__name__)   # 'factorial'
print(factorial.__code__)   # <code object factorial at ...>

# 函数属性完整列表
print(dir(factorial))
# ['__annotations__', '__call__', '__class__', '__closure__',
#  '__code__', '__defaults__', '__dict__', '__doc__', ...]
```

## 3. 高阶函数

```python
# =============================================
# 高阶函数：接受函数为参数，或把函数作为返回值
# =============================================

fruits = ['strawberry', 'fig', 'apple', 'cherry', 'raspberry', 'banana']

# ---- sorted：最常用的高阶函数 ----
# 按单词长度排序
print(sorted(fruits, key=len))
# ['fig', 'apple', 'cherry', 'banana', 'raspberry', 'strawberry']

# 按反向拼写排序（押韵词典）
print(sorted(fruits, key=lambda word: word[::-1]))
# ['banana', 'apple', 'fig', 'raspberry', 'strawberry', 'cherry']

# ---- map 和 filter 的现代替代品 ----

# 传统：map + filter
result = list(map(factorial, filter(lambda n: n % 2, range(6))))
# [1, 6, 120]（奇数的阶乘）

# ✅ 推荐：列表推导式（更易读）
result = [factorial(n) for n in range(6) if n % 2]
# [1, 6, 120]

# ---- reduce 的现代替代品 ----
from functools import reduce
from operator import add

# 传统：reduce 求和
print(reduce(add, range(100)))    # 4950

# ✅ 推荐：内置 sum 函数
print(sum(range(100)))            # 4950

# ---- all 和 any（短路归约函数）----
print(all([1, 2, 3]))    # True（所有元素为真）
print(all([1, 0, 3]))    # False（0 是假值）
print(all([]))           # True（空序列）

print(any([0, 0.0, 7]))  # True（7 是真值，短路停止）
print(any([]))           # False（空序列）

# 短路演示：any 在遇到真值时立即停止
g = (n for n in [0, 0.0, 7, 8])
print(any(g))      # True（停在 7）
print(next(g))     # 8（7 之后的下一个，7 本身被 any 消耗了）
```

## 4. 匿名函数（lambda）

```python
# =============================================
# lambda：使用 Python 表达式创建匿名函数
# 限制：主体只能是表达式，不能有语句
# =============================================

# ---- 最适合场景：高阶函数的参数 ----
fruits = ['strawberry', 'fig', 'apple', 'cherry', 'raspberry', 'banana']

# 用 lambda 定义排序键
sorted_by_reverse = sorted(fruits, key=lambda word: word[::-1])

# ---- lambda 等价于 def 语句 ----
# 以下两种定义完全等价：
double_lambda = lambda x: x * 2

def double_def(x):
    return x * 2

print(double_lambda(5), double_def(5))  # 10 10

# ---- Fredrik Lundh 的 lambda 重构秘诀 ----
# 遇到难以理解的 lambda，按以下步骤重构：
# 1. 写注释说明 lambda 的作用
# 2. 找一个描述注释的名称
# 3. 把 lambda 换成 def 语句，用那个名称命名
# 4. 删除注释

# ❌ 难以理解的 lambda
sorted_fruits = sorted(fruits, key=lambda s: s[::-1])

# ✅ 重构后（意图更清晰）
def reversed_spelling(word):
    """返回单词的反向拼写（用于押韵排序）。"""
    return word[::-1]

sorted_fruits = sorted(fruits, key=reversed_spelling)

# ⚠️ lambda 的限制：
# - 不能有赋值语句（=），但可以用 := 海象运算符
# - 不能有 while/try/return/yield 等语句
# - 适合简单的一行逻辑，复杂逻辑应该定义命名函数
```

## 5. 9 种可调用对象

```python
# =============================================
# 判断对象能否调用：callable() 内置函数
# =============================================

print([callable(obj) for obj in (abs, str, 'Ni!')])
# [True, True, False]

# Python 3.9+ 的 9 种可调用对象：

# 1. 用户定义的函数
def my_func(): pass
print(callable(my_func))          # True

# 2. 内置函数（C 实现）
print(callable(len))              # True

# 3. 内置方法（C 实现）
print(callable([].append))        # True

# 4. 方法（类中定义的函数）
class MyClass:
    def method(self): pass
print(callable(MyClass.method))   # True

# 5. 类（调用类 → 创建实例）
print(callable(MyClass))          # True

# 6. 类的实例（实现了 __call__）
class CallableInstance:
    def __call__(self): return '被调用了'

obj = CallableInstance()
print(callable(obj))              # True
print(obj())                      # '被调用了'

# 7. 生成器函数（含 yield 的函数）
def gen_func():
    yield 1

print(callable(gen_func))         # True（生成器函数是可调用对象）
print(callable(gen_func()))       # False（生成器对象不可调用！）

# 8. 原生协程函数（async def，Python 3.5+）
import asyncio
async def coro(): pass
print(callable(coro))             # True

# 9. 异步生成器函数（async def + yield，Python 3.6+）
async def async_gen():
    yield 1
print(callable(async_gen))        # True
```

## 6. 用户定义的可调用类型（`__call__`）

```python
import random

# =============================================
# 实现 __call__：让类实例像函数一样被调用
# 用途：需要在多次调用间保持状态的"函数"
# =============================================

class BingoCage:
    """从随机排序的列表中逐个取出元素（不重复）。"""

    def __init__(self, items):
        self._items = list(items)     # 创建副本，防止副作用
        random.shuffle(self._items)   # 打乱顺序

    def pick(self):
        """取出并返回一个元素。"""
        try:
            return self._items.pop()
        except IndexError:
            raise LookupError('从空 BingoCage 中取元素')

    def __call__(self):
        """bingo() 是 bingo.pick() 的快捷方式。"""
        return self.pick()


bingo = BingoCage(range(3))
print(bingo.pick())      # 取出一个元素
print(bingo())           # 等价于 bingo.pick()（通过 __call__）
print(callable(bingo))   # True（实现了 __call__）

# ---- 适用场景 ----
# 1. 需要在调用之间维护状态（比纯闭包更清晰）
# 2. 实现装饰器（装饰器必须可调用，且常需要维护状态）
# 3. 缓存/备忘录（memoization）
```

## 7. 函数参数：位置参数、关键字参数和仅限参数

```python
# =============================================
# Python 提供极为灵活的参数处理机制
# =============================================

def tag(name, *content, class_=None, **attrs):
    """生成一个或多个 HTML 标签。

    name:    标签名（位置参数）
    *content: 标签内容（可变位置参数，捕获为元组）
    class_:  class 属性（仅限关键字参数，因为 class 是关键字）
    **attrs: 其他属性（可变关键字参数，捕获为字典）
    """
    if class_ is not None:
        attrs['class'] = class_
    attr_str = ''.join(f' {k}="{v}"' for k, v in sorted(attrs.items()))
    if content:
        tags = [f'<{name}{attr_str}>{c}</{name}>' for c in content]
        return '\n'.join(tags)
    else:
        return f'<{name}{attr_str} />'


# ---- 各种调用方式 ----
print(tag('br'))                         # <br />（单个位置参数）
print(tag('p', 'hello'))                 # <p>hello</p>
print(tag('p', 'hello', 'world'))        # <p>hello</p>\n<p>world</p>
print(tag('p', 'hello', id=33))          # <p id="33">hello</p>
print(tag('p', 'hello', class_='sidebar'))  # <p class="sidebar">hello</p>

# 使用 ** 拆包字典传入参数
my_tag = {'name': 'img', 'title': 'Sunset', 'src': 'sunset.jpg', 'class': 'framed'}
print(tag(**my_tag))  # <img class="framed" src="sunset.jpg" title="Sunset" />


# =============================================
# 仅限关键字参数（keyword-only argument）
# 位于 * 或 *args 之后的参数
# =============================================

def f(a, *, b):          # b 只能通过关键字传入
    return a, b

print(f(1, b=2))         # (1, 2) ✅
# f(1, 2)   → TypeError（b 不能按位置传入）


# =============================================
# 仅限位置参数（positional-only argument）
# Python 3.8+ 新增，位于 / 之前的参数
# =============================================

def divmod(a, b, /):     # a, b 只能按位置传入
    return (a // b, a % b)

print(divmod(10, 3))              # (3, 1) ✅
# divmod(a=10, b=3)  → TypeError（不能用关键字传入）


# =============================================
# 综合示例：所有参数类型
# =============================================

# def func(pos_only1, pos_only2, /, positional_or_kw, *, kw_only1, kw_only2):
#          ↑ 仅限位置                  ↑ 普通参数                ↑ 仅限关键字
#          （/ 之前）              （/ 和 * 之间）             （* 之后）
```

## 8. operator 模块

```python
from functools import reduce
from operator import (mul, add, itemgetter, attrgetter,
                      methodcaller)

# =============================================
# operator 模块：将运算符封装为函数
# 替代 lambda，让函数式编程更简洁
# =============================================

# ---- 算术运算符函数 ----

# 计算阶乘（不用 lambda）
def factorial(n):
    return reduce(mul, range(1, n + 1))   # mul 替代 lambda a, b: a*b

print(factorial(5))    # 120

# ---- itemgetter：根据索引/键提取项 ----
metro_data = [
    ('Tokyo', 'JP', 36.933),
    ('Delhi NCR', 'IN', 21.935),
    ('Mexico City', 'MX', 20.142),
]

# 按第二个字段（国家代码）排序
for city in sorted(metro_data, key=itemgetter(1)):
    print(city)
# ('Delhi NCR', 'IN', 21.935)
# ('Tokyo', 'JP', 36.933)
# ('Mexico City', 'MX', 20.142)

# itemgetter 多个索引 → 返回元组
cc_name = itemgetter(1, 0)
print([cc_name(city) for city in metro_data])
# [('JP', 'Tokyo'), ('IN', 'Delhi NCR'), ('MX', 'Mexico City')]

# ---- attrgetter：根据属性名提取属性 ----
from collections import namedtuple

LatLon = namedtuple('LatLon', 'lat lon')
Metropolis = namedtuple('Metropolis', 'name cc pop coord')

metro_areas = [Metropolis(name, cc, pop, LatLon(lat, lon))
               for name, cc, pop, (lat, lon) in [
                   ('Tokyo', 'JP', 36.933, (35.69, 139.69)),
                   ('Delhi NCR', 'IN', 21.935, (28.61, 77.21)),
               ]]

# 按纬度排序（深层属性 coord.lat）
name_lat = attrgetter('name', 'coord.lat')
for city in sorted(metro_areas, key=attrgetter('coord.lat')):
    print(name_lat(city))
# ('Delhi NCR', 28.61)
# ('Tokyo', 35.69)

# ---- methodcaller：调用方法 ----
s = 'The time has come'
upcase = methodcaller('upper')
print(upcase(s))               # 'THE TIME HAS COME'

hyphenate = methodcaller('replace', ' ', '-')
print(hyphenate(s))            # 'The-time-has-come'

# operator 模块完整函数列表（54 个）：
# abs, add, and_, attrgetter, concat, contains, countOf, delitem,
# eq, floordiv, ge, getitem, gt, iadd, iand, iconcat, ifloordiv,
# ilshift, imatmul, imod, imul, index, indexOf, inv, invert, ior,
# ipow, irshift, is_, is_not, isub, itemgetter, itruediv, ixor,
# le, length_hint, lshift, lt, matmul, methodcaller, mod, mul,
# ne, neg, not_, or_, pos, pow, rshift, setitem, sub, truediv,
# truth, xor
```

## 9. `functools.partial` 冻结参数

```python
from operator import mul
from functools import partial

# =============================================
# partial：把多参数函数改成更少参数的可调用对象
# 冻结某些参数，生成新的可调用对象
# =============================================

# ---- 基本用法 ----
triple = partial(mul, 3)      # 冻结第一个位置参数为 3
print(triple(7))              # 21（相当于 mul(3, 7)）
print(list(map(triple, range(1, 10))))
# [3, 6, 9, 12, 15, 18, 21, 24, 27]

# ---- 实用示例：Unicode 规范化 ----
import unicodedata, functools

# 把 unicodedata.normalize 的第一个参数固定为 'NFC'
nfc = functools.partial(unicodedata.normalize, 'NFC')

s1 = 'café'
s2 = 'cafe\u0301'      # NFD 形式（两个码点）
print(s1 == s2)        # False（码点序列不同）
print(nfc(s1) == nfc(s2))  # True（NFC 规范化后相等）

# ---- 冻结位置参数和关键字参数 ----
def tag(name, *content, class_=None, **attrs):
    # 省略实现...
    pass

# 冻结位置参数 'img' 和关键字参数 class_='pic-frame'
picture = partial(tag, 'img', class_='pic-frame')
result = picture(src='wumpus.jpeg')
# 等价于 tag('img', src='wumpus.jpeg', class_='pic-frame')

# partial 对象的属性
print(picture.func)         # <function tag at ...>（原函数）
print(picture.args)         # ('img',)（冻结的位置参数）
print(picture.keywords)     # {'class_': 'pic-frame'}（冻结的关键字参数）

# ---- partialmethod 用于方法 ----
from functools import partialmethod

class Cell:
    def __init__(self):
        self._alive = False

    @property
    def alive(self):
        return self._alive

    def set_state(self, state):
        self._alive = bool(state)

    # 用 partialmethod 创建便捷方法
    set_alive = partialmethod(set_state, True)
    set_dead  = partialmethod(set_state, False)


c = Cell()
c.set_alive()
print(c.alive)    # True
c.set_dead()
print(c.alive)    # False
```

## 10. 常见误区与陷阱

### ❌ 误区一：lambda 中使用语句

```python
# ❌ lambda 不能含语句（赋值、try/except、while 等）
# wrong = lambda x: (result = x * 2)   → SyntaxError

# ✅ 复杂逻辑用 def 定义命名函数
def process(x):
    result = x * 2
    return result
```

### ❌ 误区二：以为生成器函数调用后立即执行

```python
def gen():
    print('开始执行')
    yield 1

g = gen()      # 调用生成器函数，返回生成器对象，但不执行函数体！
print('生成器对象已创建')
next(g)        # 现在才执行到第一个 yield → 打印'开始执行'
```

### ❌ 误区三：忘记 `reduce` 在 Python 3 中移到了 `functools`

```python
# ❌ Python 3 中 reduce 不再是内置函数
# reduce(add, range(100))   → NameError

# ✅ 需要导入
from functools import reduce
from operator import add
print(reduce(add, range(100)))    # 4950

# 或者更简单地用 sum：
print(sum(range(100)))            # 4950
```

### ❌ 误区四：`itemgetter` 和 `attrgetter` 的参数含义

```python
from operator import itemgetter, attrgetter

# itemgetter(1) → 取索引 1 的元素（适用于序列和映射）
get_second = itemgetter(1)
print(get_second([10, 20, 30]))  # 20（索引 1）
print(get_second({'a': 1, 1: 'b'}))  # 'b'（键 1）

# attrgetter('coord.lat') → 取 obj.coord.lat（支持嵌套属性）
# 这是 lambda obj: obj.coord.lat 的简洁替代
```

### ❌ 误区五：以为 `callable()` 能检测所有可调用对象类型

```python
# callable() 检查是否有 __call__ 方法
# 但它无法区分不同类型的可调用对象

def func(): pass
gen_func = lambda: (x for x in [])()  # 这是普通函数

print(callable(func))       # True
print(callable(len))        # True（内置函数）
print(callable(int))        # True（类）

# 要区分类型，用 type() 或 isinstance()
import types
print(isinstance(func, types.FunctionType))  # True（用户定义函数）
print(isinstance(len, types.BuiltinFunctionType))  # True（内置函数）
```
