# 第10章：使用一等函数实现设计模式

> **核心思想**：在有一等函数的语言中，某些面向对象设计模式可以简化甚至消除。策略模式、命令模式等"单方法类"模式，可以直接用函数替代，代码更简洁、维护更容易。

## 2. 案例：重构策略模式

**问题描述**：电商折扣系统有三种折扣规则：
- 积分 ≥ 1000 → 订单享 5% 折扣
- 单个商品数量 ≥ 20 → 该商品享 10% 折扣
- 订单不同商品种类 ≥ 10 → 整单享 7% 折扣

**策略模式定义**（《设计模式》）：
> 定义一系列算法，把它们一一封装起来，并且使它们可以相互替换。本模式使得算法可以独立于使用它的客户而变化。

## 3. 经典策略模式（类实现）

```python
from abc import ABC, abstractmethod
from collections.abc import Sequence
from decimal import Decimal
from typing import NamedTuple, Optional


class Customer(NamedTuple):
    name: str
    fidelity: int   # 积分


class LineItem(NamedTuple):
    product: str
    quantity: int
    price: Decimal

    def total(self) -> Decimal:
        return self.price * self.quantity


class Order(NamedTuple):
    """上下文（Context）：使用策略计算折扣。"""
    customer: Customer
    cart: Sequence[LineItem]
    promotion: Optional['Promotion'] = None   # 策略接口

    def total(self) -> Decimal:
        totals = (item.total() for item in self.cart)
        return sum(totals, start=Decimal(0))

    def due(self) -> Decimal:
        if self.promotion is None:
            discount = Decimal(0)
        else:
            discount = self.promotion.discount(self)
        return self.total() - discount

    def __repr__(self):
        return f'<Order total: {self.total():.2f} due: {self.due():.2f}>'


class Promotion(ABC):
    """策略（Strategy）：抽象基类。"""
    @abstractmethod
    def discount(self, order: Order) -> Decimal:
        """返回折扣金额（正值）。"""


class FidelityPromo(Promotion):
    """具体策略1：积分 ≥ 1000 → 5% 折扣。"""
    def discount(self, order: Order) -> Decimal:
        rate = Decimal('0.05')
        if order.customer.fidelity >= 1000:
            return order.total() * rate
        return Decimal(0)


class BulkItemPromo(Promotion):
    """具体策略2：单个商品数量 ≥ 20 → 10% 折扣。"""
    def discount(self, order: Order) -> Decimal:
        discount = Decimal(0)
        for item in order.cart:
            if item.quantity >= 20:
                discount += item.total() * Decimal('0.1')
        return discount


class LargeOrderPromo(Promotion):
    """具体策略3：不同商品种类 ≥ 10 → 7% 折扣。"""
    def discount(self, order: Order) -> Decimal:
        distinct_items = {item.product for item in order.cart}
        if len(distinct_items) >= 10:
            return order.total() * Decimal('0.07')
        return Decimal(0)


# ---- 使用示例 ----
joe = Customer('John Doe', 0)
ann = Customer('Ann Smith', 1100)
cart = [
    LineItem('banana', 4, Decimal('.5')),
    LineItem('apple', 10, Decimal('1.5')),
    LineItem('watermelon', 5, Decimal(5)),
]

print(Order(joe, cart, FidelityPromo()))
# <Order total: 42.00 due: 42.00>（joe 积分不足）

print(Order(ann, cart, FidelityPromo()))
# <Order total: 42.00 due: 39.90>（ann 享 5% 折扣）

# 问题：每次都要实例化策略类
# 这些类只有一个方法，没有状态，像函数多于像类
```

## 4. 函数实现策略模式

```python
from dataclasses import dataclass
from decimal import Decimal
from typing import Optional, Callable, NamedTuple
from collections.abc import Sequence


class Customer(NamedTuple):
    name: str
    fidelity: int


class LineItem(NamedTuple):
    product: str
    quantity: int
    price: Decimal

    def total(self):
        return self.price * self.quantity


@dataclass(frozen=True)
class Order:
    """上下文：策略现在是可调用对象（函数），而不是类实例。"""
    customer: Customer
    cart: Sequence[LineItem]
    # 关键改变：promotion 是可调用对象或 None
    promotion: Optional[Callable[['Order'], Decimal]] = None

    def total(self) -> Decimal:
        totals = (item.total() for item in self.cart)
        return sum(totals, start=Decimal(0))

    def due(self) -> Decimal:
        if self.promotion is None:
            discount = Decimal(0)
        else:
            # 调用可调用对象 self.promotion，传入 self
            # 注意：这里有两个 self——第一个获取属性，第二个传参数
            discount = self.promotion(self)
        return self.total() - discount

    def __repr__(self):
        return f'<Order total: {self.total():.2f} due: {self.due():.2f}>'


# ---- 策略：普通函数（不再是类！）----

def fidelity_promo(order: Order) -> Decimal:
    """积分 ≥ 1000 → 5% 折扣。"""
    if order.customer.fidelity >= 1000:
        return order.total() * Decimal('0.05')
    return Decimal(0)


def bulk_item_promo(order: Order) -> Decimal:
    """单个商品数量 ≥ 20 → 10% 折扣。"""
    discount = Decimal(0)
    for item in order.cart:
        if item.quantity >= 20:
            discount += item.total() * Decimal('0.1')
    return discount


def large_order_promo(order: Order) -> Decimal:
    """不同商品种类 ≥ 10 → 7% 折扣。"""
    distinct_items = {item.product for item in order.cart}
    if len(distinct_items) >= 10:
        return order.total() * Decimal('0.07')
    return Decimal(0)


# ---- 使用示例（更简洁！）----
joe = Customer('John Doe', 0)
ann = Customer('Ann Smith', 1100)
cart = [
    LineItem('banana', 4, Decimal('.5')),
    LineItem('apple', 10, Decimal('1.5')),
    LineItem('watermelon', 5, Decimal(5)),
]

# 直接传函数对象（无须实例化策略类）
print(Order(joe, cart, fidelity_promo))
# <Order total: 42.00 due: 42.00>

print(Order(ann, cart, fidelity_promo))
# <Order total: 42.00 due: 39.90>

# ---- 优势 ----
# 1. 无须定义抽象基类 Promotion
# 2. 无须实例化具体策略类（函数只创建一次，天然共享）
# 3. 代码量大幅减少
# 4. 函数就是轻量级的"可共享对象"（相当于享元模式）
```

## 5. 选择最佳策略：`globals()` 内省

```python
# =============================================
# 问题：如何选择折扣最大的策略？
# 同时：如何避免手动维护策略列表？
# =============================================

# ---- 方法1：手动维护策略列表 ----
promos = [fidelity_promo, bulk_item_promo, large_order_promo]

def best_promo(order: Order) -> Decimal:
    """选择折扣最大的策略。"""
    return max(promo(order) for promo in promos)

# 问题：添加新策略时，需要同时更新 promos 列表！

# ---- 方法2：使用 globals() 自动发现策略 ----
# globals() 返回当前模块全局命名空间的字典
# 所有以 '_promo' 结尾的函数都是策略（约定命名）
promos = [promo for name, promo in globals().items()
          if name.endswith('_promo')
          and name != 'best_promo']   # 排除 best_promo 本身

def best_promo(order: Order) -> Decimal:
    """自动从全局命名空间发现所有策略函数。"""
    return max(promo(order) for promo in promos)

# 现在只需定义新的 *_promo 函数，best_promo 自动发现！

# ---- 方法3：从单独模块发现策略 ----
import inspect
import promotions   # 假设所有策略在 promotions 模块中

promos = [func for _, func in inspect.getmembers(promotions, inspect.isfunction)]

def best_promo(order: Order) -> Decimal:
    return max(promo(order) for promo in promos)
```

## 6. 使用装饰器改进策略注册

```python
# =============================================
# 最优雅的方案：用装饰器显式注册策略
# 优点：
# 1. 明确标记（不依赖命名约定）
# 2. 策略可以放在任何模块
# 3. 无须内省，无须约定
# =============================================

Promotion = Callable[[Order], Decimal]

promos: list[Promotion] = []   # 策略注册表


def promotion(promo: Promotion) -> Promotion:
    """装饰器：把函数注册到 promos 列表，并原样返回。"""
    promos.append(promo)
    return promo   # 返回原函数（不修改，只注册）


def best_promo(order: Order) -> Decimal:
    """选择折扣最大的已注册策略。"""
    return max(promo(order) for promo in promos)


# ---- 使用装饰器注册策略 ----
@promotion      # 自动注册到 promos 列表
def fidelity_promo(order: Order) -> Decimal:
    """积分 ≥ 1000 → 5% 折扣。"""
    if order.customer.fidelity >= 1000:
        return order.total() * Decimal('0.05')
    return Decimal(0)


@promotion      # 自动注册
def bulk_item_promo(order: Order) -> Decimal:
    """单个商品数量 ≥ 20 → 10% 折扣。"""
    discount = Decimal(0)
    for item in order.cart:
        if item.quantity >= 20:
            discount += item.total() * Decimal('0.1')
    return discount


@promotion      # 自动注册
def large_order_promo(order: Order) -> Decimal:
    """不同商品种类 ≥ 10 → 7% 折扣。"""
    distinct_items = {item.product for item in order.cart}
    if len(distinct_items) >= 10:
        return order.total() * Decimal('0.07')
    return Decimal(0)


# 添加新策略只需：1. 定义函数 2. 加 @promotion 装饰器
# 不需要修改 best_promo，不需要维护列表！
print(promos)  # [<function fidelity_promo...>, ...]
```

## 7. 命令模式

```python
# =============================================
# 命令模式：把操作封装成对象
# 同样可以用函数替代只有一个方法的类
# =============================================

# ---- 经典命令模式（类实现）----
class MacroCommand:
    """执行一系列命令的命令。"""
    def __init__(self, commands):
        self.commands = list(commands)   # 命令列表

    def __call__(self):
        for command in self.commands:
            command()   # 执行每个命令


# 示例命令
class OpenCommand:
    def __call__(self):
        print('打开文件')

class SaveCommand:
    def __call__(self):
        print('保存文件')

macro = MacroCommand([OpenCommand(), SaveCommand()])
macro()
# 打开文件
# 保存文件

# ---- 函数实现（更简洁）----
# 不需要 Command 类！直接用函数作为命令

def open_file():
    print('打开文件')

def save_file():
    print('保存文件')

# 宏命令就是函数列表 + 一个执行循环
macro_commands = [open_file, save_file]

def execute_macro(commands):
    for cmd in commands:
        cmd()

execute_macro(macro_commands)

# ---- 带状态的命令（闭包替代实例变量）----
def make_remove_command(file_path: str):
    """工厂函数：创建删除特定文件的命令（闭包保存状态）。"""
    def remove():
        import os
        os.remove(file_path)   # file_path 保存在闭包中
        print(f'删除文件: {file_path}')
    return remove

rm_temp = make_remove_command('/tmp/tempfile.txt')
# rm_temp()   # 执行：删除 /tmp/tempfile.txt
```

## 8. 常见误区与陷阱

### ❌ 误区一：添加新策略但忘记注册到列表

```python
# ❌ 用 globals() 方法时：只定义函数，但命名不符合约定
def seasonal_discount(order):   # ← 没有 '_promo' 后缀！
    return order.total() * Decimal('0.03')

# best_promo 不会发现这个策略！

# ✅ 使用装饰器注册：无此问题
@promotion
def seasonal_promo(order):    # ← 装饰器自动注册
    return order.total() * Decimal('0.03')
```

### ❌ 误区二：策略含有状态时仍用函数

```python
# ❌ 如果策略需要维护调用次数、缓存等状态，简单函数不够
def fidelity_promo_with_cache(order: Order) -> Decimal:
    # 想缓存计算结果，但函数没有状态...
    pass

# ✅ 有状态的策略应该用类
class FidelityPromoWithCache:
    def __init__(self):
        self._cache = {}   # 缓存

    def __call__(self, order: Order) -> Decimal:
        key = id(order)
        if key not in self._cache:
            # 计算折扣...
            self._cache[key] = Decimal(0)
        return self._cache[key]
```

### ❌ 误区三：数据类滥用（把行为分散到外部）

```python
# ❌ 数据类只有数据，行为散落在系统各处（代码异味）
@dataclass
class Order:
    items: list
    customer: str
    discount: float = 0.0

# 相关行为散落到处：
def calculate_total(order):   # 行为应该在 Order 里！
    return sum(...)

# ✅ 把行为放回类中
@dataclass
class Order:
    items: list
    customer: str
    discount: float = 0.0

    def total(self) -> float:          # 行为属于 Order
        return sum(item.price for item in self.items)

    def due(self) -> float:
        return self.total() * (1 - self.discount)
```
