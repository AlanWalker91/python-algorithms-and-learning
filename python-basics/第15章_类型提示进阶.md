## 2. 重载的函数签名（`@overload`）

```python
# =============================================
# @overload：为灵活的函数提供多个精确的签名
# 让类型检查工具根据参数推断返回类型
# =============================================

import functools
import operator
from collections.abc import Iterable
from typing import overload, Union, TypeVar

T = TypeVar('T')
S = TypeVar('S')

# ---- 重载的签名（类型检查用，不执行）----
@overload
def sum(it: Iterable[T]) -> Union[T, int]: ...

@overload
def sum(it: Iterable[T], /, start: S) -> Union[T, S]: ...

# ---- 实际实现（无类型提示）----
def sum(it, /, start=0):
    return functools.reduce(operator.add, it, start)


# 类型检查工具根据调用方式选择匹配的签名：
result1 = sum([1, 2, 3])          # 匹配第1个签名，返回 int
result2 = sum([1, 2, 3], 10)      # 匹配第2个签名，返回 int
result3 = sum(['a', 'b'], 'z')    # 匹配第2个签名，返回 str


# =============================================
# 内置 max 函数的 6 个重载签名
# =============================================

from typing import Protocol, Any

class SupportsLessThan(Protocol):
    """支持 < 运算符的类型协议。"""
    def __lt__(self, other: Any) -> bool: ...


LT = TypeVar('LT', bound=SupportsLessThan)
DT = TypeVar('DT')

# 6个重载涵盖所有参数组合：
# 1. 多个 LT 参数，无 key 和 default
@overload
def max(__arg1: LT, __arg2: LT, *args: LT, key: None = ...) -> LT: ...

# 2. 多个 T 参数，有 key 函数
@overload
def max(__arg1: T, __arg2: T, *args: T,
        key: Callable[[T], LT]) -> T: ...

# 3. 可迭代 LT，无 key 和 default
@overload
def max(__iterable: Iterable[LT], *, key: None = ...) -> LT: ...

# 4. 可迭代 T，有 key 函数
@overload
def max(__iterable: Iterable[T], *, key: Callable[[T], LT]) -> T: ...

# 5. 可迭代 LT，有 default（可能返回 default）
@overload
def max(__iterable: Iterable[LT], *, key: None = ...,
        default: DT) -> Union[LT, DT]: ...

# 6. 可迭代 T，有 key 和 default
@overload
def max(__iterable: Iterable[T], *, key: Callable[[T], LT],
        default: DT) -> Union[T, DT]: ...


# @overload 的关键规则：
# 1. 重载签名放在实现前面
# 2. 每个重载签名用 ... 作为函数体
# 3. 实现本身不需要类型提示
# 4. 类型检查工具按顺序匹配，选第一个匹配的
```

## 3. TypedDict：为字典记录添加类型

```python
# =============================================
# TypedDict：为用作记录的字典添加类型提示
# 注意：运行时无效，仅供类型检查工具使用
# =============================================

from typing import TypedDict


class BookDict(TypedDict):
    """图书记录的类型定义。

    TypedDict 的两个作用：
    1. 使用类语法为字典的"字段"添加类型提示
    2. 告诉类型检查工具字典应有哪些键和值类型
    """
    isbn: str
    title: str
    authors: list[str]
    pagecount: int


# ---- 运行时行为（就是普通 dict！）----
book = BookDict(
    isbn='978-0-13-475759-9',
    title='Refactoring, 2e',
    authors=['Martin Fowler', 'Kent Beck'],
    pagecount=478
)

print(type(book))    # <class 'dict'>（不是 BookDict！）
print(book['title']) # 'Refactoring, 2e'
# book.title         # AttributeError（dict 没有属性访问）

# ---- TypedDict 的限制 ----
# 1. 不能有默认值（不能 pagecount: int = 0）
# 2. 不能定义方法
# 3. 运行时不检查类型（可以放入错误类型的值）
wrong_book = BookDict(isbn=123, title='test', authors='Bob', pagecount=0)
# 不报错！只有 Mypy 等工具会发现类型错误

# ---- TypedDict vs 数据类构建器 ----
# TypedDict：仅供类型检查，运行时无任何功能
# NamedTuple：运行时有效，创建不可变对象，有方法
# @dataclass：运行时有效，创建可变或不可变对象，可有方法


# ---- 使用 TypedDict 注解函数 ----
AUTHOR_ELEMENT = '<AUTHOR>{}</AUTHOR>'

def to_xml(book: BookDict) -> str:
    """将图书记录转换为 XML 字符串。"""
    elements: list[str] = []
    for key, value in book.items():
        if isinstance(value, list):
            elements.extend(AUTHOR_ELEMENT.format(n) for n in value)
        else:
            tag = key.upper()
            elements.append(f'<{tag}>{value}</{tag}>')
    xml = '\n\t'.join(elements)
    return f'<BOOK>\n\t{xml}\n</BOOK>'
```

## 4. 运行时访问类型提示

```python
# =============================================
# 类型提示在运行时的存储和访问
# =============================================

# ---- __annotations__ 属性 ----
class Demo:
    a: int         # 只有注解，没有值
    b: float = 1.1 # 注解 + 类属性
    c = 'spam'     # 类属性，无注解

print(Demo.__annotations__)
# {'a': int, 'b': float}（只有注解的才在这里）
# c 不在里面（没有注解）

# ---- 运行时注解的问题 ----
from __future__ import annotations  # 推迟注解求值（PEP 563）

class ForwardRef:
    def method(self) -> 'ForwardRef':   # 前向引用（字符串形式）
        return self

# 没有 __future__ 时，self -> 'ForwardRef' 在类定义时就要解析
# 用了 __future__ 后，所有注解推迟为字符串，只在需要时解析

# ---- typing.get_type_hints() ----
import typing

class Coordinate:
    lat: float
    lon: float

# 在有 from __future__ import annotations 时，
# __annotations__ 里存的是字符串；get_type_hints() 解析为实际类型
print(Coordinate.__annotations__)      # {'lat': 'float', 'lon': 'float'}（字符串）
print(typing.get_type_hints(Coordinate))  # {'lat': float, 'lon': float}（类型对象）
```

## 5. 实现泛化类（`Generic[T]`）

```python
from typing import Generic, TypeVar

T = TypeVar('T')

# =============================================
# Generic[T]：参数化类（支持类型变量的类）
# =============================================

class LottoBlower(Generic[T]):
    """通用抽奖机，可以存储任意类型的元素。"""

    def __init__(self, items: Iterable[T]) -> None:
        self._balls = list(items)
        random.shuffle(self._balls)

    def pick(self) -> T:
        """随机取出一个元素，返回类型与存入类型相同。"""
        try:
            return self._balls.pop()
        except IndexError:
            raise LookupError('抽奖机是空的')

    def loaded(self) -> bool:
        return bool(self._balls)

    def inspect(self) -> tuple[T, ...]:
        return tuple(sorted(self._balls))


# 使用时指定类型参数：
machine = LottoBlower[int](range(1, 50))  # LottoBlower[int]
print(machine.pick())   # 某个 int

str_machine = LottoBlower[str](['a', 'b', 'c'])
print(str_machine.pick())   # 某个 str

# 类型检查工具能推断出：
item = machine.pick()   # item 的类型是 int
# item = str_machine.pick()   # 类型是 str
```

## 6. 型变：协变、逆变与不变

```python
# =============================================
# 型变（Variance）：泛型类型与其类型参数子类型关系
# =============================================

# ---- 背景：类型关系 ----
# int 是 float 的子类型（int <: float）
# 但是 list[int] 是 list[float] 的子类型吗？

# 不是！list 是不变的（invariant）
# 如果 list[int] <: list[float]，那么可以：
def add_float(container: list[float]) -> None:
    container.append(3.14)  # 向 list[float] 中加入 float

int_list: list[int] = [1, 2, 3]
# add_float(int_list)  # 如果这是合法的，则 int_list 会包含 float！
# 这违反了类型安全，所以 list 是不变的

# =============================================
# 1. 不变（invariant）：默认行为
# =============================================

# class BeverageDispenser(Generic[T]):
#     "T 是不变的：只接受精确的类型"

# =============================================
# 2. 协变（covariant）：可以用更具体（子类）的类型
# TypeVar(covariant=True) 或 TypeVar('T_co', covariant=True)
# =============================================

T_co = TypeVar('T_co', covariant=True)

class ReadOnlyBeverageDispenser(Generic[T_co]):
    """只读（不能修改内容），协变。
    如果 Juice <: Beverage，则
    ReadOnlyBeverageDispenser[Juice] <: ReadOnlyBeverageDispenser[Beverage]
    """
    def __init__(self, beverage: T_co) -> None:
        self._beverage = beverage

    def dispense(self) -> T_co:
        return self._beverage


# =============================================
# 3. 逆变（contravariant）：可以用更抽象（父类）的类型
# TypeVar(contravariant=True)
# =============================================

T_contra = TypeVar('T_contra', contravariant=True)

class TrashCan(Generic[T_contra]):
    """垃圾桶：可以扔进去（输入），逆变。
    如果 Compostable <: Biodegradable，则
    TrashCan[Biodegradable] <: TrashCan[Compostable]
    （能接收更多类型的垃圾桶更"有用"）
    """
    def dispose(self, item: T_contra) -> None:
        pass


# =============================================
# 型变规则（经验法则）
# =============================================
# - 只输出 T（只有 __get__/return T）→ 协变
# - 只输入 T（只有参数类型是 T）→ 逆变
# - 既输入又输出 T（读和写）→ 不变（默认）


# ---- 型变总结 ----
# 协变（out，"生产者"）：
#   ImmutableBag[Juice] <: ImmutableBag[Beverage]
# 逆变（in，"消费者"）：
#   TrashCan[Biodegradable] <: TrashCan[Compostable]
# 不变（in + out）：
#   list[int] 与 list[float] 没有子类型关系
```

## 7. 实现泛化静态协议

```python
from typing import Protocol, TypeVar, runtime_checkable

# =============================================
# Protocol[T]：参数化协议（泛化静态协议）
# =============================================

LT = TypeVar('LT', bound='SupportsLessThan')


class SupportsLessThan(Protocol):
    """任何实现了 __lt__ 的类型。"""
    def __lt__(self, other: Any) -> bool: ...


class SupportsAbs(Protocol[T_co]):
    """参数化协议：__abs__ 返回类型参数 T_co。"""
    def __abs__(self) -> T_co: ...


# 使用参数化协议：
import math
from typing import TypeVar, NamedTuple

ST = TypeVar('ST', bound=SupportsAbs)

def manhattan_distance(x: SupportsAbs[float], y: SupportsAbs[float]) -> float:
    return abs(x) + abs(y)


# 更完整的示例：max 函数使用 SupportsLessThan
T = TypeVar('T')
LT_T = TypeVar('LT_T', bound=SupportsLessThan)

def max_sorted(items: Iterable[LT_T]) -> LT_T:
    """返回可迭代对象中的最大值（要求元素支持 < 运算）。"""
    it = iter(items)
    try:
        result = next(it)
    except StopIteration:
        raise ValueError('max() arg is an empty sequence')
    for item in it:
        if result < item:
            result = item
    return result

# 类型检查工具能验证：
max_sorted([1, 2, 3])          # ✅ int 支持 <
max_sorted(['a', 'b', 'c'])    # ✅ str 支持 <
# max_sorted([[1], [2]])       # ❌ Mypy 警告（list 不支持 <）
```

## 8. 常见误区与陷阱

### ❌ 误区一：以为 TypedDict 有运行时类型检查

```python
from typing import TypedDict

class Point(TypedDict):
    x: float
    y: float

# ❌ TypedDict 在运行时不检查类型！
p = Point(x='hello', y=None)   # 不报错！
print(p)   # {'x': 'hello', 'y': None}

# ✅ 需要运行时验证？用 pydantic 或 @dataclass + __post_init__
```

### ❌ 误区二：@overload 签名会执行

```python
# ❌ 误区：@overload 签名是"多个实现"
@overload
def process(x: int) -> int: ...

@overload
def process(x: str) -> str: ...

def process(x):   # 只有这个实现才会被调用！
    return x

# @overload 仅用于类型检查工具，实际调用的是最后那个没有 @overload 的实现
```

### ❌ 误区三：混淆协变和逆变

```python
# 记忆技巧（Liskov 替换原则）：
# - 协变（"out"/"生产者"）：输出类型可以更具体（子类）
#   → 如果你只读数据，子类型的容器可以替代父类型的容器
#   → TypeVar(covariant=True) 或 out_T
# - 逆变（"in"/"消费者"）：输入类型可以更宽泛（父类）
#   → 如果你只写数据，父类型的容器可以替代子类型的容器
#   → TypeVar(contravariant=True) 或 in_T

# 简单口诀：
# 只读 → 协变（子类型可替代）
# 只写 → 逆变（父类型可替代）
# 读写 → 不变（类型必须精确匹配）
```

### ❌ 误区四：泛化类忘记继承 `Generic[T]`

```python
from typing import TypeVar, Generic

T = TypeVar('T')

# ❌ 没有继承 Generic[T]，类型检查工具无法追踪类型参数
class Box:
    def __init__(self, item: T) -> None:
        self.item = item

    def get(self) -> T:
        return self.item

# ✅ 正确：继承 Generic[T]
class Box(Generic[T]):
    def __init__(self, item: T) -> None:
        self.item = item

    def get(self) -> T:
        return self.item

int_box = Box(42)
item = int_box.get()   # 类型检查工具知道 item 是 int
```
