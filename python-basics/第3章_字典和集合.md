## 2. 字典的现代句法

### 2.1 字典推导式

```python
# 从可迭代对象的键值对构建 dict（自 Python 2.7+）

dial_codes = [
    (880, 'Bangladesh'),
    (55, 'Brazil'),
    (86, 'China'),
    (91, 'India'),
    (7,  'Russia'),
    (1,  'United States'),
]

# 基本字典推导式（对调键值）
country_dial = {country: code for code, country in dial_codes}
print(country_dial)
# {'Bangladesh': 880, 'Brazil': 55, ..., 'United States': 1}

# 带条件和转换的字典推导式
filtered = {code: country.upper()
            for country, code in sorted(country_dial.items())
            if code < 70}
print(filtered)  # {55: 'BRAZIL', 62: 'INDONESIA', 7: 'RUSSIA', 1: 'UNITED STATES'}
```

### 2.2 映射拆包（`**`）

```python
# Python 3.5+ 支持在函数调用和 dict 字面量中多次使用 **

# 函数调用中的多次 **（键必须是字符串且唯一）
def dump(**kwargs):
    return kwargs

result = dump(**{'x': 1}, y=2, **{'z': 3})
print(result)  # {'x': 1, 'y': 2, 'z': 3}

# dict 字面量中的 **（允许键重复，后面的覆盖前面的）
merged = {'a': 0, **{'x': 1}, 'y': 2, **{'z': 3, 'x': 4}}
print(merged)  # {'a': 0, 'x': 4, 'y': 2, 'z': 3}  ← x 被覆盖为 4
```

### 2.3 使用 `|` 和 `|=` 合并映射（Python 3.9+）

```python
d1 = {'a': 1, 'b': 3}
d2 = {'a': 2, 'b': 4, 'c': 6}

# | 创建新映射（d1 不变）
print(d1 | d2)   # {'a': 2, 'b': 4, 'c': 6}（d2 覆盖 d1 的重复键）
print(d1)        # {'a': 1, 'b': 3}（不变）

# |= 就地更新 d1
d1 |= d2
print(d1)        # {'a': 2, 'b': 4, 'c': 6}（d1 已更新）
```

## 3. 使用模式匹配处理映射

```python
# match/case 的匹配对象可以是映射（collections.abc.Mapping 的子类）
# 映射模式：部分匹配成功（不需要匹配所有键）

def get_creators(record: dict) -> list:
    """从出版物记录中提取创作者的名字。

    演示映射模式匹配：
    - 模式中键的顺序无关紧要（OrderedDict 也能匹配）
    - 只需匹配指定的键，额外的键（如 'title'）不影响匹配
    - 使用 **details 捕获多出的键值对
    """
    match record:
        case {'type': 'book', 'api': 2, 'authors': [*names]}:
            # 匹配含 type='book', api=2，且 authors 是序列的映射
            return names
        case {'type': 'book', 'api': 1, 'author': name}:
            # 匹配含 type='book', api=1，且 author 是单个值的映射
            return [name]
        case {'type': 'book'}:
            # 其他含 type='book' 的映射
            raise ValueError(f"Invalid 'book' record: {record!r}")
        case {'type': 'movie', 'director': name}:
            return [name]
        case _:
            raise ValueError(f'Invalid record: {record!r}')


# 测试
b1 = dict(api=1, author='Guido', type='book', title='Python')
print(get_creators(b1))   # ['Guido']

# **extra 捕获额外键值对
food = dict(category='ice cream', flavor='vanilla', cost=199)
match food:
    case {'category': 'ice cream', **details}:
        print(f'Ice cream details: {details}')
        # {'flavor': 'vanilla', 'cost': 199}
```

## 4. 映射类型的标准 API

### 4.1 使用 `setdefault` 更新可变值

```python
import re, sys

WORD_RE = re.compile(r'\w+')
index = {}

# ❌ 不够高效：需要对 index 搜索两次
for line_no, line in enumerate(open(sys.argv[1]), 1):
    for match in WORD_RE.finditer(line):
        word = match.group()
        location = (line_no, match.start() + 1)
        occurrences = index.get(word, [])   # 第一次搜索
        occurrences.append(location)
        index[word] = occurrences            # 第二次搜索

# ✅ 推荐：setdefault 只搜索一次（获取 + 设置 + 返回）
for line_no, line in enumerate(open(sys.argv[1]), 1):
    for match in WORD_RE.finditer(line):
        word = match.group()
        location = (line_no, match.start() + 1)
        # 如果 word 不存在，设为 []；返回现有列表（直接 append）
        index.setdefault(word, []).append(location)

# setdefault(key, []) 等价于：
# if key not in my_dict:
#     my_dict[key] = []
# return my_dict[key]
# 但 setdefault 只做一次键查找
```

## 5. 可哈希（Hashable）是什么

```python
# Python 术语表定义：
# 如果一个对象的哈希码在整个生命周期内永不改变（依托 __hash__），
# 而且可与其他对象比较（依托 __eq__），那么这个对象就是可哈希的。
# 两个可哈希对象仅当哈希码相同时相等。

# 数值类型、str、bytes 是可哈希的
hash(42)        # 42
hash('hello')   # 某个整数
hash(b'bytes')  # 某个整数

# 不可变容器：只有所有项均可哈希时才可哈希
tt = (1, 2, (30, 40))
hash(tt)         # 正常

tl = (1, 2, [30, 40])   # 含列表！
hash(tl)         # TypeError: unhashable type: 'list'

tf = (1, 2, frozenset([30, 40]))  # frozenset 可哈希
hash(tf)         # 正常

# 用户自定义类：默认可哈希（哈希码来自 id()）
class MyClass: pass
hash(MyClass())  # 正常

# 自定义了 __eq__ 但没有 __hash__：Python 将 __hash__ 设为 None → 不可哈希
class OnlyEq:
    def __eq__(self, other): return True
# hash(OnlyEq())  → TypeError: unhashable type: 'OnlyEq'

# 同时定义 __eq__ 和 __hash__：基于不变属性，保证相等对象哈希码相同
class Point:
    def __init__(self, x, y):
        self.x = x
        self.y = y
    def __eq__(self, other):
        return self.x == other.x and self.y == other.y
    def __hash__(self):
        return hash((self.x, self.y))  # 只用不变属性
```

## 6. 自动处理缺失键：`setdefault` 和 `defaultdict`

### 6.1 `collections.defaultdict`

```python
import collections, re, sys

WORD_RE = re.compile(r'\w+')

# 创建 defaultdict，default_factory=list（list 是可调用对象）
index = collections.defaultdict(list)

for line_no, line in enumerate(open(sys.argv[1]), 1):
    for match in WORD_RE.finditer(line):
        word = match.group()
        location = (line_no, match.start() + 1)
        # 如果 word 不存在，自动创建空列表并关联
        index[word].append(location)

# defaultdict 的工作原理：
# dd = defaultdict(list)
# dd['new-key']  ← 触发 dd.__missing__('new-key')
#                 ← 调用 default_factory()（即 list()）
#                 ← 把 [] 存入 dd['new-key']
#                 ← 返回 []

# ⚠️ 注意：default_factory 只为 __getitem__ 调用
# dd.get('missing')   → None（不触发 default_factory！）
# 'missing' in dd     → False（不触发！）
```

## 7. `__missing__` 方法

```python
import collections


class StrKeyDict0(dict):
    """查找时把非字符串键转换成字符串的字典。

    继承 dict：演示 __missing__ 的工作机制。
    实际项目应继承 UserDict（见 StrKeyDict）。
    """

    def __missing__(self, key):
        """当 __getitem__ 找不到键时调用（不是 get 或 in！）。

        关键：先判断 key 是否已经是 str，防止无限递归。
        """
        if isinstance(key, str):
            # 如果 key 已经是字符串但还是找不到，真的不存在
            raise KeyError(key)
        # 否则转换为 str 再查找
        return self[str(key)]   # ← 重新触发 __getitem__，再次查找

    def get(self, key, default=None):
        """委托给 __getitem__（让 __missing__ 有机会处理）。"""
        try:
            return self[key]    # ← 通过 __getitem__ 触发 __missing__
        except KeyError:
            return default

    def __contains__(self, key):
        """同时检查原始键和字符串形式的键。"""
        return key in self.keys() or str(key) in self.keys()


# 测试
d = StrKeyDict0([('2', 'two'), ('4', 'four')])
print(d['2'])     # 'two'（直接找到）
print(d[4])       # 'four'（4 转为 '4' 后找到）
print(d.get(4))   # 'four'
print(2 in d)     # True（str(2)='2' 存在）
print(1 in d)     # False


class StrKeyDict(collections.UserDict):
    """继承 UserDict 的简洁版本——推荐做法。

    比 dict 子类更简洁：
    - UserDict 是纯 Python 实现，不会忽略覆盖的方法
    - __contains__ 更简洁（内部存储已转为 str）
    - __setitem__ 自动把键转为 str
    """

    def __missing__(self, key):
        if isinstance(key, str):
            raise KeyError(key)
        return self[str(key)]

    def __contains__(self, key):
        # 所有键都以 str 存储，直接检查 self.data
        return str(key) in self.data

    def __setitem__(self, key, item):
        # 存储时把键转为 str
        self.data[str(key)] = item
```

## 8. dict 的变体

### 8.1 `collections.OrderedDict`（Python 3.7+ 不常用）

```python
from collections import OrderedDict

# 自 Python 3.7 起，普通 dict 也保留插入顺序
# OrderedDict 的特殊之处：
# 1. 等值检查考虑顺序（dict 不考虑）
# 2. popitem(last=False) 删除最先插入的项（先进先出）
# 3. move_to_end(key) 把键移到开头或末尾

od = OrderedDict()
od['a'] = 1
od['b'] = 2
od['c'] = 3
od.move_to_end('a')         # 'a' 移到末尾
print(list(od))              # ['b', 'c', 'a']
od.move_to_end('c', last=False)  # 'c' 移到开头
print(list(od))              # ['c', 'b', 'a']

# 等值比较考虑顺序
print(OrderedDict([('a', 1), ('b', 2)]) ==
      OrderedDict([('b', 2), ('a', 1)]))  # False（顺序不同）
print(dict([('a', 1), ('b', 2)]) ==
      dict([('b', 2), ('a', 1)]))          # True（dict 不看顺序）
```

### 8.2 `collections.ChainMap`

```python
from collections import ChainMap

# ChainMap 存放多个映射的引用，作为一个整体搜索
d1 = dict(a=1, b=3)
d2 = dict(a=2, b=4, c=6)

chain = ChainMap(d1, d2)
print(chain['a'])    # 1（在 d1 中找到，停止搜索）
print(chain['c'])    # 6（d1 没有，在 d2 中找到）

# 更新只影响第一个映射
chain['c'] = -1
print(d1)    # {'a': 1, 'b': 3, 'c': -1}（c 写入 d1）
print(d2)    # {'a': 2, 'b': 4, 'c': 6}（d2 不变）

# 典型用途：模拟 Python 的变量查找作用域
import builtins
pylookup = ChainMap(locals(), globals(), vars(builtins))
```

### 8.3 `collections.Counter`

```python
from collections import Counter

# 计数映射：更新现有键时计数增加
ct = Counter('abracadabra')
print(ct)                        # Counter({'a': 5, 'b': 2, 'r': 2, 'c': 1, 'd': 1})
ct.update('aaaaazzz')
print(ct)                        # Counter({'a': 10, 'z': 3, ...})
print(ct.most_common(3))         # [('a', 10), ('z', 3), ('b', 2)]

# Counter 支持算术运算（合并计数）
c1 = Counter('aab')
c2 = Counter('bbc')
print(c1 + c2)   # Counter({'b': 3, 'a': 2, 'c': 1})
print(c1 - c2)   # Counter({'a': 2})（负计数不保留）
```

### 8.4 `collections.UserDict`（自定义映射的最佳基类）

```python
import collections


class MyMapping(collections.UserDict):
    """继承 UserDict 创建自定义映射。

    推荐理由：
    - UserDict 用 Python 实现，不走内置 C 语言捷径
    - 覆盖的方法（__setitem__ 等）总是会被调用
    - 比继承 dict 代码更少、更安全

    内部使用 self.data（一个普通 dict）存储实际数据。
    """

    def __missing__(self, key):
        # 自定义缺键行为
        return f'Default for {key}'

    def __setitem__(self, key, item):
        # 自定义写入（例如：类型检查）
        if not isinstance(key, str):
            raise TypeError(f'Key must be str, not {type(key)}')
        self.data[key] = item    # 写入内部 dict
```

from types import MappingProxyType

# 创建 dict 的只读代理
d = {'a': 1, 'b': 2}

## 10. 字典视图

```python
d = dict(a=10, b=20, c=30)

# 三种视图：keys()、values()、items()
# 均为动态代理——原 dict 修改后视图立即更新
keys   = d.keys()    # dict_keys
values = d.values()  # dict_values
items  = d.items()   # dict_items

# 视图的共同特性
print(len(keys))        # 3
print(list(keys))       # ['a', 'b', 'c']
print(reversed(values)) # 迭代器

# ⚠️ 视图不支持索引
# keys[0]   → TypeError: 'dict_keys' object is not subscriptable

# 修改原 dict，视图立即更新
d['z'] = 99
print(values)    # dict_values([10, 20, 30, 99])

# dict_keys 和 dict_items 支持集合运算（类似 frozenset）
d1 = dict(a=1, b=2, c=3, d=4)
d2 = dict(b=20, d=40, e=50)

# 两个字典都有的键（交集）
print(d1.keys() & d2.keys())   # {'b', 'd'}

# 字典视图支持与 set 的运算
s = {'a', 'e', 'i'}
print(d1.keys() & s)    # {'a'}
print(d1.keys() | s)    # {'a', 'c', 'b', 'd', 'i', 'e'}

# ⚠️ dict_items 只在所有值均可哈希时才能用作集合
# 如果 dict 中有不可哈希的值（如列表），dict_items 不能参与集合运算
```

## 11. dict 的哈希表实现对实践的影响

```python
# 1. 键必须可哈希
d = {}
# d[[1, 2]] = 'val'   → TypeError: unhashable type: 'list'
d[(1, 2)] = 'val'    # ✅ 元组可哈希

# 2. 通过键访问极快（O(1)）
# 哈希表通过计算哈希码直接定位，而不是遍历

# 3. CPython 3.7+ 保留插入顺序（3.6 以前无序！）
d = {'a': 1, 'b': 2, 'c': 3}
print(list(d))  # ['a', 'b', 'c']（保留插入顺序）

# 4. 占用内存较多（哈希表需保留 1/3 空位）
# → 不要用 dict 存储大量小对象（用 __slots__ 或 namedtuple 代替）

# 5. 不要在 __init__ 之外创建实例属性（破坏键共享优化）
# Python 3.3+ 同类实例可共享哈希表，但在 __init__ 外添加属性会破坏这种优化

# 6. 最佳实践：在 __init__ 中一次性设置所有实例属性
class GoodClass:
    def __init__(self, a, b):
        self.a = a  # 在 __init__ 中设置
        self.b = b

class BadClass:
    def __init__(self, a, b):
        self.a = a
    def add_b(self, b):
        self.b = b  # 在 __init__ 之外添加 → 破坏键共享优化
```

## 12. 集合：set 和 frozenset

### 12.1 集合的基本用法

```python
# ---- 创建集合 ----
# set 字面量（推荐，速度更快）
s1 = {1, 2, 3}

# 构造函数（必须用于创建空集合）
empty = set()   # ⚠️ {} 创建的是空 dict，不是空 set！

# frozenset（不可变，可哈希，可作为集合元素或字典键）
fs = frozenset([1, 2, 3])

# ---- 集合的核心功能：去重 ----
l = ['spam', 'spam', 'eggs', 'spam']
unique = set(l)          # {'eggs', 'spam'}
print(list(unique))      # ['eggs', 'spam']

# 去重并保留顺序（Python 3.7+ dict 保留插入顺序）
unique_ordered = list(dict.fromkeys(l))  # ['spam', 'eggs']

# ---- 集合推导式 ----
from unicodedata import name
# 构建包含"SIGN"字符名的 Latin-1 字符集合
signs = {chr(i) for i in range(32, 256) if 'SIGN' in name(chr(i), '')}
# → {'§', '=', '¢', '#', '¤', '<', '¥', 'μ', '×', '$', ...}

# ---- 高效成员测试 ----
haystack = {i for i in range(10**6)}   # 100万个元素的集合
needles = {1, 2, 3, 100, 200, 999}

# 集合交集（O(1) 成员测试，远快于循环）
found = len(needles & haystack)         # 一行代码
print(found)  # 6

# 等价但较慢的循环
found = sum(1 for n in needles if n in haystack)
```

### 12.2 `frozenset` 的特殊用途

```python
# frozenset 是不可变的，可哈希
fs = frozenset([1, 2, 3])
hash(fs)   # 正常

# set 可以包含 frozenset 作为元素
s = {frozenset([1, 2]), frozenset([3, 4])}

# frozenset 可以作为字典键
d = {frozenset([1, 2]): 'pair'}
```

## 13. 集合运算总览

```python
a = {1, 2, 3, 4, 5}
b = {3, 4, 5, 6, 7}

# ---- 数学集合运算（运算符版，要求两个操作数都是集合）----
print(a & b)   # 交集：{3, 4, 5}
print(a | b)   # 并集：{1, 2, 3, 4, 5, 6, 7}
print(a - b)   # 差集（a 有 b 没有）：{1, 2}
print(a ^ b)   # 对称差集（只在其中一个里）：{1, 2, 6, 7}

# ---- 方法版（可接受任意可迭代对象作为参数）----
print(a.intersection(b))         # {3, 4, 5}
print(a.union(b))                # {1, 2, 3, 4, 5, 6, 7}
print(a.difference(b))           # {1, 2}
print(a.symmetric_difference(b)) # {1, 2, 6, 7}

# 方法版接受可迭代对象（不只是集合）
print(a.intersection([3, 4, 5, 8]))   # {3, 4, 5}

# ---- 就地更新（仅 set，frozenset 不支持）----
a &= b   # a 更新为与 b 的交集
a |= b   # a 更新为与 b 的并集
a -= b   # a 更新为与 b 的差集

# ---- 集合谓词（返回布尔值）----
print({1, 2}.issubset({1, 2, 3}))    # True（子集）
print({1, 2, 3}.issuperset({1, 2}))  # True（超集）
print({1, 2}.isdisjoint({3, 4}))     # True（不相交）
print(2 in {1, 2, 3})                # True（成员测试）
```

### 集合与字典视图的集合运算

```python
d1 = dict(a=1, b=2, c=3)
d2 = dict(b=20, d=40, e=50)

# dict_keys 支持集合运算（类似 frozenset）
common_keys = d1.keys() & d2.keys()   # {'b'}
all_keys = d1.keys() | d2.keys()      # {'a', 'b', 'c', 'd', 'e'}

# dict_items 也支持（前提是所有值可哈希）
common_items = d1.items() & d2.items()  # 空集（值不同）
```

## 14. 常见误区与陷阱

### ❌ 误区一：用 `{}` 创建空集合

```python
# ❌ 错误！{} 是空字典，不是空集合
empty = {}
print(type(empty))   # <class 'dict'>

# ✅ 正确：创建空集合用 set()
empty_set = set()
print(type(empty_set))   # <class 'set'>
```

### ❌ 误区二：以为 `d.get(k)` 会触发 `__missing__`

```python
from collections import defaultdict

dd = defaultdict(list)
dd['existing'] = [1, 2, 3]

# ❌ 误区：以为 get() 也会触发 default_factory
result = dd.get('new_key')   # None（不触发 default_factory！）
print('new_key' in dd)       # False（也不触发！）

# ✅ 只有 dd[key] 才会触发 default_factory 和 __missing__
result = dd['new_key']       # [] （触发了 default_factory）
```

### ❌ 误区三：修改字典时迭代

```python
d = {'a': 1, 'b': 2, 'c': 3}

# ❌ 运行时错误：迭代时修改字典大小
# for key in d:
#     if key == 'a':
#         del d[key]   → RuntimeError: dictionary changed size during iteration

# ✅ 先收集要删除的键，再删除
to_delete = [key for key in d if key == 'a']
for key in to_delete:
    del d[key]
```

### ❌ 误区四：把含可变对象的元组用作字典键

```python
# ❌ 含列表的元组不可哈希，不能作为字典键
t = (1, [2, 3])
d = {}
# d[t] = 'value'   → TypeError: unhashable type: 'list'

# ✅ 只有全部元素可哈希的元组才能作为键
t_ok = (1, (2, 3))   # 嵌套元组，可哈希
d[t_ok] = 'value'    # ✅
```

### ❌ 误区五：以为 `Counter` 中缺失键返回 `KeyError`

```python
from collections import Counter

ct = Counter('abcabc')
print(ct['a'])   # 2
print(ct['z'])   # 0（不抛 KeyError！Counter 对缺失键返回 0）
print('z' in ct) # False（'z' 未统计过）
```
