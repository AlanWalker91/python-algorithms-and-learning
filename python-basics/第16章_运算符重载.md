# 第16章：运算符重载

> **核心思想**：运算符重载让用户自定义类型能使用中缀运算符表达式，使代码更易读。Python 在灵活性、可用性和安全性方面取得了很好的平衡：不能改变内置类型运算符的含义，不能新建运算符，通过 `NotImplemented` 实现分派机制。

## 2. 运算符重载入门

```python
# =============================================
# Python 运算符重载的三条限制（安全性保证）
# =============================================

# 1. 不能改变内置类型运算符的含义
# 1 + 2 永远是 3，不能让它变成别的

# 2. 不能新建运算符（如 Java 的 =>）
# 只能重载 Python 已有的运算符

# 3. 以下运算符不能重载：
# is（同一性）、and（逻辑与）、or（逻辑或）、not（逻辑非）
# 但按位运算符可以重载：&（__and__）、|（__or__）、~（__invert__）

# ---- 优势：让用户定义类型更自然 ----
# Python 版（运算符重载）：
# interest = principal * ((1 + rate) ** periods - 1)

# Java 版（BigDecimal，不支持运算符重载）：
# BigDecimal interest = principal.multiply(
#     BigDecimal.ONE.add(rate).pow(periods).subtract(BigDecimal.ONE));
```

## 3. 一元运算符

```python
import math

class Vector:
    def __init__(self, components):
        self._components = list(components)

    def __iter__(self):
        return iter(self._components)

    def __abs__(self):
        """支持 abs(v)，返回向量的模。"""
        return math.hypot(*self)

    def __neg__(self):
        """支持 -v（一元取反），返回各分量取反的新向量。

        ⚠️ 不修改 self！运算符应该始终返回新对象。
        """
        return Vector(-x for x in self)

    def __pos__(self):
        """支持 +v（一元取正），返回 self 的副本。

        通常 +x == x，但有例外（见"x 和 +x 何时不相等"）。
        """
        return Vector(self)   # 不可变时可以 return self

    # __invert__ 对 Vector 没有明确含义，不实现
    # 用户试图 ~v 时，Python 会抛出 TypeError


# ---- x 和 +x 何时不相等 ----

# 例1：Decimal 精度变化
import decimal
ctx = decimal.getcontext()
ctx.prec = 40
one_third = decimal.Decimal('1') / decimal.Decimal('3')
print(one_third == +one_third)   # True（精度相同）
ctx.prec = 28
print(one_third == +one_third)   # False！
# +one_third 用当前精度 28 重新计算，而 one_third 还是 40 位精度

# 例2：Counter 正号去掉零值和负值
from collections import Counter
ct = Counter('abracadabra')
ct['r'] = -3
ct['d'] = 0
print(ct)    # Counter({'a': 5, 'b': 2, 'c': 1, 'd': 0, 'r': -3})
print(+ct)   # Counter({'a': 5, 'b': 2, 'c': 1})（零值和负值被移除）
```

## 4. 重载加法运算符 +（中缀运算符分派）

```python
import itertools

class Vector:
    def __init__(self, components):
        self._components = list(components)

    def __iter__(self):
        return iter(self._components)

    def __add__(self, other):
        """重载 v1 + v2，返回新向量（各分量之和）。

        使用 zip_longest 支持不同长度的向量（短的用 0 补齐）。
        捕获 TypeError 返回 NotImplemented，让解释器尝试反向操作。
        """
        try:
            pairs = itertools.zip_longest(self, other, fillvalue=0.0)
            return Vector(a + b for a, b in pairs)
        except TypeError:
            return NotImplemented   # ← 返回 NotImplemented，不是抛出！

    def __radd__(self, other):
        """反向加法：当左操作数不支持 + 时调用。

        最简单的实现：委托给 __add__（利用交换律）。
        """
        return self + other   # 等价于 self.__add__(other)
        # 或者：return __add__(other)


# =============================================
# 中缀运算符分派机制（以 a + b 为例）
# =============================================

# 1. 调用 a.__add__(b)
#    - 返回结果 → 使用该结果
#    - 返回 NotImplemented → 继续步骤2

# 2. 调用 b.__radd__(a)
#    - 返回结果 → 使用该结果
#    - 返回 NotImplemented → 抛出 TypeError

# 关键：有"反向"版本（radd、rsub、rmul 等）
# __radd__ 让 10 + v1 也能工作（int 不知道如何处理 Vector）


# 使用示例
v1 = Vector([3, 4, 5])
v2 = Vector([6, 7, 8])

print(v1 + v2)       # Vector([9.0, 11.0, 13.0])
print(v1 + (10, 20)) # Vector([13.0, 24.0, 5.0])（元组填充为 0）
print((10,) + v1)    # 通过 v1.__radd__((10,)) 实现


# =============================================
# NotImplemented vs NotImplementedError
# =============================================

# NotImplemented：特殊的单例值（不是异常）
# - 运算符方法在无法处理给定操作数时返回它
# - 让解释器尝试调用反向方法
# - 不能用 raise NotImplemented（这是错的！）

# NotImplementedError：异常（TypeError 的子类）
# - 抽象方法中使用（表示子类必须实现）
# - 用 raise NotImplementedError 抛出

from abc import ABC, abstractmethod

class AbstractVector(ABC):
    @abstractmethod
    def __add__(self, other):
        raise NotImplementedError   # 抽象类中用异常


class ConcreteVector(AbstractVector):
    def __add__(self, other):
        try:
            pairs = itertools.zip_longest(self, other, fillvalue=0.0)
            return Vector(a + b for a, b in pairs)
        except TypeError:
            return NotImplemented   # 运算符方法中返回单例值
```

## 5. 重载标量乘法 *

```python
class Vector:
    def __mul__(self, scalar):
        """向量乘以标量：v * 3。

        使用鸭子类型：尝试将 scalar 转为 float，
        失败则返回 NotImplemented（而非 isinstance 检查）。
        """
        try:
            factor = float(scalar)   # 接受 int、float、Fraction 等
        except TypeError:
            return NotImplemented
        return Vector(n * factor for n in self)

    def __rmul__(self, scalar):
        """反向：3 * v（标量乘以向量）。"""
        return self * scalar   # 委托给 __mul__


# 使用示例
v = Vector([1, 2, 3])
print(v * 10)           # Vector([10.0, 20.0, 30.0])
print(10 * v)           # 通过 __rmul__：Vector([10.0, 20.0, 30.0])
print(True * v)         # Vector([1.0, 2.0, 3.0])（bool 是 int 子类）

from fractions import Fraction
print(v * Fraction(1, 3))  # Vector([0.333..., 0.666..., 1.0])

try:
    v * 'abc'            # str 不能转 float → NotImplemented → TypeError
except TypeError as e:
    print(e)
```

## 6. 使用 `@` 作为中缀运算符（矩阵乘法）

```python
# =============================================
# Python 3.5 引入 @ 运算符（PEP 465）
# 用于矩阵乘法（NumPy 等库使用）
# __matmul__、__rmatmul__
# =============================================

class Vector:
    def __matmul__(self, other):
        """计算两个向量的点积：v1 @ v2。

        使用鸭子类型：尝试迭代 other，失败返回 NotImplemented。
        """
        if (isinstance(other, abc.Sized) and
                isinstance(other, abc.Iterable)):
            if len(self) == len(other):
                return sum(a * b for a, b in zip(self, other))
            else:
                raise ValueError('向量长度不一致')
        return NotImplemented

    def __rmatmul__(self, other):
        return self @ other  # 委托给 __matmul__


# 使用示例
v1 = Vector([1, 2, 3])
v2 = Vector([4, 5, 6])

print(v1 @ v2)   # 1*4 + 2*5 + 3*6 = 32（点积）
print(v2 @ v1)   # 32（交换律）

# NumPy 大量使用 @ 做矩阵乘法：
# import numpy as np
# a = np.array([[1, 2], [3, 4]])
# b = np.array([[5, 6], [7, 8]])
# print(a @ b)  # [[19, 22], [43, 50]]
```

## 7. 比较运算符

```python
# =============================================
# 比较运算符的特殊规则
# =============================================

class Vector:
    def __eq__(self, other):
        """相等性比较（== ）。

        反向方法：==（自身！不是 !=）
        所以只需实现 __eq__，不需要 __req__。
        """
        if isinstance(other, Vector):
            return (len(self) == len(other) and
                    all(a == b for a, b in zip(self, other)))
        return NotImplemented

    def __ne__(self, other):
        """不等于（!=）。

        Python 自动从 __eq__ 派生：not (self == other)
        通常不需要手动实现 __ne__。
        """
        result = self.__eq__(other)
        if result is NotImplemented:
            return result
        return not result

    def __lt__(self, other):
        """小于（<）。
        
        反向方法是 __gt__（对调操作数）。
        a < b → a.__lt__(b)；若 NotImplemented → b.__gt__(a)
        """
        if isinstance(other, Vector):
            return abs(self) < abs(other)
        return NotImplemented

    def __le__(self, other):
        """小于等于（<=）。
        反向方法是 __ge__。
        """
        return self < other or self == other


# =============================================
# 比较运算符的反向方法对应关系
# =============================================
# a == b：a.__eq__(b)  → b.__eq__(a)
# a != b：a.__ne__(b)  → b.__ne__(a)
# a <  b：a.__lt__(b)  → b.__gt__(a)
# a <= b：a.__le__(b)  → b.__ge__(a)
# a >  b：a.__gt__(b)  → b.__lt__(a)
# a >= b：a.__ge__(b)  → b.__le__(a)

# ⚠️ == 的反向是 ==（不是 !=！）
# 这与 + 等算术运算符不同

# ---- 使用 functools.total_ordering 简化 ----
from functools import total_ordering

@total_ordering
class OrderedVector:
    """只需实现 __eq__ 和一个比较方法，其余自动生成。"""

    def __init__(self, magnitude):
        self.magnitude = magnitude

    def __eq__(self, other):
        if isinstance(other, OrderedVector):
            return self.magnitude == other.magnitude
        return NotImplemented

    def __lt__(self, other):
        if isinstance(other, OrderedVector):
            return self.magnitude < other.magnitude
        return NotImplemented
    # total_ordering 自动生成 __le__, __gt__, __ge__
```

## 8. 增量赋值运算符

```python
# =============================================
# 增量赋值运算符（如 +=、*=）
# 可变对象：实现 __iadd__ 就地修改
# 不可变对象：Python 退而使用 __add__ + 重新绑定
# =============================================

# ---- 可变对象：就地修改（ID 不变）----
l = [1, 2, 3]
print(id(l))     # 某个 ID
l += [4, 5]      # 调用 l.__iadd__([4, 5]) → 就地修改
print(l)         # [1, 2, 3, 4, 5]
print(id(l))     # 同一个 ID（对象未改变）

# ---- 不可变对象：创建新对象（ID 改变）----
t = (1, 2, 3)
print(id(t))     # 某个 ID
t += (4, 5)      # 元组无 __iadd__，退而 t = t + (4, 5) 创建新对象
print(t)         # (1, 2, 3, 4, 5)
print(id(t))     # 不同 ID（新对象）

# ---- 为可变 Vector 实现 __iadd__ ----
class MutableVector:
    def __init__(self, components):
        self._components = list(components)

    def __iadd__(self, other):
        """就地加法（+=）。
        修改 self，返回 self（⚠️ 必须返回 self！）
        """
        if isinstance(other, MutableVector):
            # 两个向量长度必须相同
            if len(self) != len(other):
                raise ValueError('向量长度不一致')
            for i, val in enumerate(other):
                self._components[i] += val
            return self   # ← 必须返回 self！
        return NotImplemented

    def __len__(self):
        return len(self._components)

    def __iter__(self):
        return iter(self._components)


v = MutableVector([1, 2, 3])
original_id = id(v)
v += MutableVector([10, 20, 30])
print(v._components)    # [11, 22, 33]
print(id(v) == original_id)  # True（就地修改，ID 不变）

# ⚠️ __iadd__ 必须返回 self！
# a += b 在 Python 内部是：a = a.__iadd__(b)
# 如果 __iadd__ 返回 None，a 会变成 None！
```

## 9. 常见误区与陷阱

### ❌ 误区一：返回 `NotImplementedError` 而不是 `NotImplemented`

```python
# ❌ 错误：抛出异常（终止运算符分派）
class Bad:
    def __add__(self, other):
        if not isinstance(other, Bad):
            raise NotImplementedError   # ← 不对！这终止了分派链！

# ✅ 正确：返回 NotImplemented（让解释器继续尝试反向操作）
class Good:
    def __add__(self, other):
        if not isinstance(other, Good):
            return NotImplemented   # ← 正确！解释器会尝试 other.__radd__(self)
```

### ❌ 误区二：在运算符方法中修改 self

```python
# ❌ 错误：修改了操作数
class Bad:
    def __add__(self, other):
        self.x += other.x    # 修改了 self！
        return self          # 错误！

# ✅ 正确：始终返回新对象
class Good:
    def __add__(self, other):
        return Good(self.x + other.x)  # 创建新对象

# 例外：__iadd__ 等就地运算符应该修改 self 并返回 self
```

### ❌ 误区三：忘记实现反向方法

```python
# ❌ 只实现 __add__，导致 2 + v 不工作
class Vector:
    def __add__(self, other):   # v + 2 ✅
        ...
    # 没有 __radd__！

v = Vector([1, 2])
v + 2     # ✅ 调用 v.__add__(2)
# 2 + v   # ❌ int.__add__(v) 返回 NotImplemented，
           # 然后找 v.__radd__(2)，但不存在 → TypeError

# ✅ 添加反向方法
class Vector:
    def __add__(self, other): ...
    def __radd__(self, other):
        return self + other  # 委托给 __add__
```

### ❌ 误区四：`__iadd__` 忘记返回 `self`

```python
# ❌ 错误：没有返回 self
class Bad:
    def __iadd__(self, other):
        self.x += other.x
        # 忘记 return self！

b = Bad()
b.x = 0
b += another   # b 现在是 None！
# 因为 b = b.__iadd__(another)，而 __iadd__ 返回了 None

# ✅ 正确：必须返回 self
class Good:
    def __iadd__(self, other):
        self.x += other.x
        return self   # ← 必须！
```
