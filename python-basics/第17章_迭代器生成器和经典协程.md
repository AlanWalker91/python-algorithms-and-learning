## 2. 序列可迭代的原因：`iter` 函数

```python
# =============================================
# 内置函数 iter() 的工作流程
# =============================================

# iter(x) 执行的步骤：
# 1. 检查对象是否有 __iter__，有则调用，获取迭代器
# 2. 没有 __iter__，但有 __getitem__（从 0 开始），
#    创建一个迭代器，尝试按索引获取项
# 3. 两者都没有 → TypeError

# ---- 只有 __getitem__ 也能迭代 ----
class Spam:
    def __getitem__(self, i):
        print(f'--> {i}')
        raise IndexError()   # 返回 IndexError 表示序列结束

spam_can = Spam()
print(iter(spam_can))    # <iterator object at ...>
print(list(spam_can))    # []（IndexError 立即触发）

# 但无法通过 ABC 检查！
from collections import abc
print(isinstance(spam_can, abc.Iterable))  # False（没有 __iter__）

# ---- 检测对象是否可迭代的最准确方式 ----
def is_iterable(obj):
    try:
        iter(obj)
        return True
    except TypeError:
        return False

print(is_iterable([1, 2]))   # True
print(is_iterable(42))       # False

# ---- iter() 的第二种用法：哨符迭代 ----
import random

def d6():
    return random.randint(1, 6)

# 掷骰子直到出现 1（哨符）
d6_iter = iter(d6, 1)
for roll in d6_iter:
    print(roll)   # 打印结果，直到掷出 1（不打印 1）

# 实用场景：按块读取文件
from functools import partial
with open('mydata.bin', 'rb') as f:
    read_block = partial(f.read, 64)   # 每次读 64 字节
    for block in iter(read_block, b''):  # 空字节串为哨符
        process(block)
```

## 3. 可迭代对象 vs 迭代器

```python
# =============================================
# 关键区别
# =============================================

# 可迭代对象（Iterable）：
# - 有 __iter__ 方法，每次调用返回新迭代器
# - 可以多次迭代

# 迭代器（Iterator）：
# - 有 __next__ 和 __iter__ 方法
# - __iter__ 返回 self（迭代器本身就是可迭代的）
# - 耗尽后不可重置（要重新迭代，需重新构建迭代器）

# for 循环的内部机制：
s = 'ABC'
# 等价于：
it = iter(s)            # 获取迭代器
while True:
    try:
        char = next(it) # 获取下一项
        print(char)
    except StopIteration:
        del it           # 释放迭代器
        break

# ---- 迭代器协议的 ABC 定义 ----
from collections import abc
print(issubclass(list, abc.Iterable))  # True（有 __iter__）
print(issubclass(list, abc.Iterator))  # False（list 没有 __next__）

# 迭代器满足：
# - isinstance(it, abc.Iterator) 或
# - hasattr(it, '__next__') and hasattr(it, '__iter__')

# ⚠️ 迭代器也是可迭代的（__iter__ 返回 self）
# 但可迭代对象不一定是迭代器

s = 'hello'
it = iter(s)
print(isinstance(it, abc.Iterable))  # True（str_iterator 有 __iter__）
print(isinstance(it, abc.Iterator))  # True（str_iterator 有 __next__）
print(isinstance(s, abc.Iterator))   # False（str 没有 __next__）
```

## 4. Sentence 类的 5 个版本

### V1：序列协议（`__getitem__`）

```python
import re, reprlib

RE_WORD = re.compile(r'\w+')

class Sentence:
    """V1：通过序列协议实现可迭代。"""

    def __init__(self, text):
        self.text = text
        self.words = RE_WORD.findall(text)  # 一次性提取所有单词

    def __getitem__(self, index):           # 序列协议
        return self.words[index]

    def __len__(self):
        return len(self.words)

    def __repr__(self):
        return f'Sentence({reprlib.repr(self.text)})'
```

### V2：经典迭代器模式（`SentenceIterator` 类）

```python
class Sentence:
    """V2：经典迭代器设计模式。"""

    def __init__(self, text):
        self.text = text
        self.words = RE_WORD.findall(text)

    def __iter__(self):
        return SentenceIterator(self.words)  # 返回新迭代器对象


class SentenceIterator:
    """迭代器：维护迭代状态。"""

    def __init__(self, words):
        self.words = words
        self.index = 0

    def __next__(self):
        try:
            word = self.words[self.index]
        except IndexError:
            raise StopIteration()
        self.index += 1
        return word

    def __iter__(self):
        return self   # 迭代器的 __iter__ 返回自身
```

### V3：生成器函数替代迭代器类（最常用）

```python
class Sentence:
    """V3：生成器函数实现 __iter__，无须迭代器类！"""

    def __init__(self, text):
        self.text = text
        self.words = RE_WORD.findall(text)

    def __iter__(self):
        """含 yield 的方法就是生成器函数。"""
        for word in self.words:
            yield word
        # 无须 return，函数结束时生成器自动抛出 StopIteration

    def __repr__(self):
        return f'Sentence({reprlib.repr(self.text)})'
```

### V4：惰性实现（`finditer`）

```python
class Sentence:
    """V4：惰性生成器，按需产出单词。"""

    def __init__(self, text):
        self.text = text        # 不存储 words 列表！

    def __iter__(self):
        # finditer 是惰性版本（返回生成器，不是列表）
        for match in RE_WORD.finditer(self.text):
            yield match.group()   # 按需提取单词
```

### V5：生成器表达式（最简洁）

```python
class Sentence:
    """V5：生成器表达式，一行实现 __iter__。"""

    def __init__(self, text):
        self.text = text

    def __iter__(self):
        return (match.group()                         # 产出值
                for match in RE_WORD.finditer(self.text))  # 数据源

    def __repr__(self):
        return f'Sentence({reprlib.repr(self.text)})'
```

## 5. 生成器函数详解

```python
# =============================================
# 生成器函数：含 yield 的函数
# 调用 → 返回生成器对象（不执行函数体）
# next() → 执行到下一个 yield，返回产出值
# =============================================

def gen_123():
    """一个产出 3 个值的简单生成器。"""
    yield 1
    yield 2
    yield 3

print(gen_123)     # <function gen_123 at ...>（函数对象）
print(gen_123())   # <generator object gen_123 at ...>（生成器对象）

g = gen_123()
print(next(g))     # 1（执行到第一个 yield）
print(next(g))     # 2
print(next(g))     # 3
try:
    next(g)        # StopIteration（函数体结束）
except StopIteration:
    print('耗尽！')


# ---- 执行流程详解 ----
def gen_AB():
    print('start')
    yield 'A'        # 暂停，产出 'A'
    print('continue')
    yield 'B'        # 暂停，产出 'B'
    print('end.')
    # 函数体结束 → 生成器自动抛出 StopIteration

for c in gen_AB():   # 迭代
    print('-->', c)

# 输出：
# start        ← gen_AB 执行到第一个 yield
# --> A        ← for 循环打印
# continue     ← gen_AB 从第一个 yield 继续执行
# --> B        ← for 循环打印
# end.         ← gen_AB 执行到函数体结束


# ---- 生成器函数 vs 生成器表达式 ----
# 生成器函数：用 def + yield（可以有复杂逻辑）
def gen_func():
    for i in range(10):
        if i % 2 == 0:
            yield i

# 生成器表达式：(expr for var in iter if cond)（简单情况）
gen_exp = (i for i in range(10) if i % 2 == 0)

# 两者都返回生成器对象，行为相同
print(list(gen_func()))   # [0, 2, 4, 6, 8]
print(list(gen_exp))      # [0, 2, 4, 6, 8]
```

## 6. 惰性实现（`finditer` vs `findall`）

```python
import re

text = 'The time has come'
RE_WORD = re.compile(r'\w+')

# findall：及早求值（一次性返回所有匹配项的列表）
words_all = RE_WORD.findall(text)  # ['The', 'time', 'has', 'come']
print(type(words_all))   # <class 'list'>

# finditer：惰性求值（返回生成器，按需产出）
words_iter = RE_WORD.finditer(text)
print(type(words_iter))  # <class 're.Match iterator>

# 对于大型文本，finditer 节省大量内存
# 只有在需要每个单词时才计算

# ---- 生成器表达式的惰性 ----
def gen_AB():
    print('start'); yield 'A'
    print('continue'); yield 'B'
    print('end.')

# 列表推导式：及早求值
res1 = [x*3 for x in gen_AB()]
# 立即打印：start, continue, end.

for i in res1:
    print('-->', i)

# 生成器表达式：惰性求值
res2 = (x*3 for x in gen_AB())
# 什么都不打印（还没开始迭代）

for i in res2:
    print('-->', i)
# 迭代时才逐个打印：start, --> AAA, continue, --> BBB, end.
```

## 7. 等差数列生成器

```python
# =============================================
# 等差数列生成器（类似 range，支持浮点数）
# =============================================

class ArithmeticProgression:
    """等差数列类（通过 __iter__ 产出值）。"""

    def __init__(self, begin, step, end=None):
        self.begin = begin
        self.step = step
        self.end = end   # None → 无穷数列

    def __iter__(self):
        result_type = type(self.begin + self.step)   # 推断结果类型
        result = result_type(self.begin)             # 转换首项类型
        forever = self.end is None
        index = 0
        while forever or result < self.end:
            yield result
            index += 1
            result = self.begin + self.step * index  # 用 index 避免浮点累积误差


# 测试
print(list(ArithmeticProgression(0, 1, 3)))        # [0, 1, 2]
print(list(ArithmeticProgression(1, .5, 3)))       # [1.0, 1.5, 2.0, 2.5]
print(list(ArithmeticProgression(0, 1/3, 1)))      # [0.0, 0.333..., 0.666...]

from fractions import Fraction
print(list(ArithmeticProgression(0, Fraction(1, 3), 1)))
# [Fraction(0, 1), Fraction(1, 3), Fraction(2, 3)]


# ---- 用 itertools 实现（更简洁）----
import itertools

def aritprog_gen(begin, step, end=None):
    """等差数列生成器（利用 itertools）。"""
    first = type(begin + step)(begin)
    ap_gen = itertools.count(first, step)     # 无穷等差数列
    if end is None:
        return ap_gen
    return itertools.takewhile(lambda n: n < end, ap_gen)

print(list(aritprog_gen(1, .5, 3)))   # [1.0, 1.5, 2.0, 2.5]
```

## 8. `itertools` 模块：标准库生成器

```python
import itertools, operator

# =============================================
# 筛选：从输入中选出子集
# =============================================

def vowel(c): return c.lower() in 'aeiou'

# filter：只产出 predicate 为真的项
print(list(filter(vowel, 'Aardvark')))           # ['A', 'a', 'a']

# filterfalse：只产出 predicate 为假的项
print(list(itertools.filterfalse(vowel, 'Aardvark')))  # ['r', 'd', 'v', 'r', 'k']

# dropwhile：跳过 predicate 为真的前缀项，然后产出剩余全部
print(list(itertools.dropwhile(vowel, 'Aardvark')))    # ['r', 'd', 'v', 'a', 'r', 'k']

# takewhile：产出 predicate 为真的项，一旦为假就停止
print(list(itertools.takewhile(vowel, 'Aardvark')))    # ['A', 'a']

# compress：按 selector 中的布尔值筛选
print(list(itertools.compress('Aardvark', (1, 0, 1, 1, 0, 1))))  # ['A', 'r', 'd', 'a']

# islice：切片（惰性，支持任意可迭代对象）
print(list(itertools.islice('Aardvark', 4)))       # ['A', 'a', 'r', 'd']
print(list(itertools.islice('Aardvark', 4, 7)))    # ['v', 'a', 'r']

# =============================================
# 映射：对每项应用函数
# =============================================

# accumulate：累积计算
sample = [5, 4, 2, 8, 7, 6, 3, 0, 9, 1]
print(list(itertools.accumulate(sample)))           # 累计和
print(list(itertools.accumulate(sample, min)))      # 累计最小值
print(list(itertools.accumulate(sample, operator.mul)))  # 累计乘积

# enumerate：产出 (index, item) 元组
print(list(enumerate('ABC', 1)))  # [(1, 'A'), (2, 'B'), (3, 'C')]

# starmap：把每项作为参数包传给函数
print(list(itertools.starmap(operator.mul, enumerate('ABC', 1))))
# ['A', 'BB', 'CCC']

# =============================================
# 合并：从多个可迭代对象产出
# =============================================

# chain：顺序连接多个可迭代对象
print(list(itertools.chain('ABC', range(2))))   # ['A', 'B', 'C', 0, 1]

# chain.from_iterable：扁平化嵌套可迭代对象
print(list(itertools.chain.from_iterable(enumerate('ABC'))))
# [0, 'A', 1, 'B', 2, 'C']

# zip：并行迭代（最短耗尽就停止）
print(list(zip('ABC', range(5), [10, 20, 30])))
# [('A', 0, 10), ('B', 1, 20), ('C', 2, 30)]

# zip_longest：并行迭代（最长耗尽才停止，缺少时用 fillvalue）
print(list(itertools.zip_longest('ABC', range(5), fillvalue='?')))
# [('A', 0), ('B', 1), ('C', 2), ('?', 3), ('?', 4)]

# product：笛卡儿积
print(list(itertools.product('AB', range(2))))
# [('A', 0), ('A', 1), ('B', 0), ('B', 1)]

# =============================================
# 扩展：产出更多值
# =============================================

# count：从 start 开始无限计数
ct = itertools.count(1, .3)
print(list(itertools.islice(ct, 4)))  # [1, 1.3, 1.6, 1.9]

# cycle：循环重复
cy = itertools.cycle('ABC')
print(list(itertools.islice(cy, 7)))  # ['A', 'B', 'C', 'A', 'B', 'C', 'A']

# repeat：重复固定值
print(list(itertools.repeat(7, 4)))   # [7, 7, 7, 7]

# pairwise：连续重叠对（Python 3.10+）
print(list(itertools.pairwise(range(5))))  # [(0,1),(1,2),(2,3),(3,4)]

# combinations：组合（不重复）
print(list(itertools.combinations('ABC', 2)))
# [('A', 'B'), ('A', 'C'), ('B', 'C')]

# permutations：排列
print(list(itertools.permutations('ABC', 2)))
# [('A', 'B'), ('A', 'C'), ('B', 'A'), ('B', 'C'), ('C', 'A'), ('C', 'B')]

# =============================================
# 重排：groupby 和 tee
# =============================================

# groupby：按分组标准聚合（输入需按标准排序）
animals = ['rat', 'bat', 'duck', 'lion', 'eagle', 'shark', 'giraffe']
animals.sort(key=len)
for length, group in itertools.groupby(animals, len):
    print(length, '->', list(group))
# 3 -> ['rat', 'bat']
# 4 -> ['duck', 'lion']
# 5 -> ['eagle', 'shark']
# 7 -> ['giraffe']

# tee：创建 n 个独立的生成器（默认 2 个）
g1, g2 = itertools.tee('ABC')
print(next(g1))   # 'A'
print(next(g2))   # 'A'（独立！）
```

## 9. 可迭代的归约函数

```python
# =============================================
# 归约函数：把可迭代对象归约为单个值
# =============================================

# all：所有项为真则 True（短路：遇到 False 立即停止）
print(all([1, 2, 3]))     # True
print(all([1, 0, 3]))     # False（遇到 0 短路）
print(all([]))            # True（空序列）

# any：有任一为真则 True（短路：遇到 True 立即停止）
print(any([0, 0.0, 7]))   # True（遇到 7 短路）
print(any([]))            # False（空序列）

# 短路演示
g = (n for n in [0, 0.0, 7, 8])
print(any(g))             # True（迭代到 7 停止）
print(next(g))            # 8（7 之后的下一个）

# max、min、sum：常见归约
print(max([3, 1, 4, 1, 5, 9]))          # 9
print(min([3, 1, 4], key=lambda x: -x)) # 4（最大的负数最小）
print(sum(range(101)))                   # 5050

# functools.reduce：通用归约
from functools import reduce
from operator import add, mul

print(reduce(add, range(101)))   # 5050（等价于 sum）
print(reduce(mul, range(1, 6)))  # 120（5!）

# reduce 提供 initializer 防止空序列报错：
print(reduce(add, [], 0))   # 0（initializer 是 XOR/AND/OR 运算的幺元）
```

## 10. `yield from`：委托子生成器

```python
# =============================================
# yield from：把生成器的工作委托给子生成器
# Python 3.3 引入
# =============================================

# ---- 无 yield from 的写法（冗长）----
def sub_gen():
    yield 1.1
    yield 1.2

def gen():
    yield 1
    for v in sub_gen():   # 手动从子生成器中获取值
        yield v           # 再产出
    yield 2

print(list(gen()))   # [1, 1.1, 1.2, 2]

# ---- 使用 yield from（简洁）----
def gen():
    yield 1
    yield from sub_gen()  # 委托给子生成器
    yield 2

print(list(gen()))   # [1, 1.1, 1.2, 2]

# ---- 遍历树形结构的实用示例 ----
def tree():
    return 'root', [
        ('leaf1', []),
        ('leaf2', [
            ('leaf3', []),
        ]),
    ]

def flatten(item):
    """递归扁平化嵌套结构。"""
    name, children = item
    yield name
    for child in children:
        yield from flatten(child)   # 递归委托

root = tree()
print(list(flatten(root)))   # ['root', 'leaf1', 'leaf2', 'leaf3']


# ---- Sentence 中使用 yield from ----
class Sentence:
    def __init__(self, text):
        self.text = text

    def __iter__(self):
        # yield from 委托给 finditer 返回的生成器
        yield from (match.group()
                    for match in RE_WORD.finditer(self.text))
```

## 11. 经典协程简介

```python
# =============================================
# 经典协程：通过 .send() 向生成器传值
# 注意：Python 3.5+ 推荐使用 async/await 而非经典协程
# =============================================

def averager():
    """计算平均值的协程（经典协程示例）。

    - 调用方发送数值（.send(value)）
    - 协程产出当前平均值
    - 协程在 yield 处暂停和恢复
    """
    total = 0.0
    count = 0
    average = None
    while True:
        term = yield average   # 产出当前平均值，并等待接收下一个值
        total += term
        count += 1
        average = total / count


# 使用协程
coro = averager()
next(coro)           # "预激"（prime）：把协程推进到第一个 yield

result = coro.send(10)   # 发送 10，协程恢复执行，产出新平均值
print(result)            # 10.0

result = coro.send(30)
print(result)            # 20.0

result = coro.send(5)
print(result)            # 15.0

# 关闭协程
coro.close()   # 抛出 GeneratorExit 到协程中

# ⚠️ 现代 Python（3.5+）：
# - 经典协程适用于复杂控制流
# - 简单异步任务应使用 async def / await（第21章）
```

## 12. 常见误区与陷阱

### ❌ 误区一：把可迭代对象变成迭代器

```python
# ❌ 反模式：在 Sentence 本身实现 __next__
class BadSentence:
    def __init__(self, words):
        self.words = words
        self.index = 0

    def __iter__(self):
        return self   # ← 返回 self（让自身成为迭代器）

    def __next__(self):   # ← 实现 __next__（自身就是迭代器）
        try:
            word = self.words[self.index]
            self.index += 1
            return word
        except IndexError:
            raise StopIteration()

# 问题：无法多次独立迭代！
bad = BadSentence(['hello', 'world'])
it1 = iter(bad)
it2 = iter(bad)   # 两个"迭代器"实际上是同一个对象！
print(next(it1))  # 'hello'
print(next(it2))  # 'world'（不是 'hello'！状态共享了！）
```

### ❌ 误区二：以为生成器函数调用后立即执行

```python
# ❌ 误区：调用生成器函数就执行函数体
def gen():
    print('执行开始')
    yield 1

g = gen()     # 调用函数，返回生成器对象，但不执行函数体！
print('生成器已创建')
next(g)       # 现在才执行到 yield → 打印"执行开始"

# 输出：
# 生成器已创建
# 执行开始
```

### ❌ 误区三：迭代器用完后可以重置

```python
# ❌ 误区：迭代器可以重置
s = 'ABC'
it = iter(s)
print(list(it))   # ['A', 'B', 'C']
print(list(it))   # []（已耗尽！）

# ✅ 需要重新迭代，要重新调用 iter()
it = iter(s)      # 重新创建迭代器
print(list(it))   # ['A', 'B', 'C']
```

### ❌ 误区四：生成器表达式中使用延迟变量

```python
# ❌ 陷阱：闭包捕获变量引用，而不是值
funcs = [lambda: i for i in range(3)]
print([f() for f in funcs])   # [2, 2, 2]（都是最后的 i 值！）

# ✅ 解决：用默认参数捕获当前值
funcs = [lambda i=i: i for i in range(3)]
print([f() for f in funcs])   # [0, 1, 2]

# 生成器表达式无此问题（每次迭代立即产出，不存在延迟引用）
gen = (i for i in range(3))   # 不会有闭包问题
```
