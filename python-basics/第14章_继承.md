## 2. super() 函数

### 2.1 基本用法

```python
from collections import OrderedDict


class LastUpdatedOrderedDict(OrderedDict):
    """按照最后更新顺序存储项。"""

    def __setitem__(self, key, value):
        # ✅ 推荐：使用 super() 调用超类方法
        super().__setitem__(key, value)   # 插入/更新键值对
        self.move_to_end(key)             # 把这个键移到末尾


# ❌ 不推荐：硬编码基类名称
class NotRecommended(OrderedDict):
    def __setitem__(self, key, value):
        OrderedDict.__setitem__(self, key, value)  # 硬编码了基类名！
        self.move_to_end(key)
        # 缺陷1：如果基类名称改变，这里也要改
        # 缺陷2：在多重继承中无法支持协作调用链


# 覆盖 __init__ 时，务必调用 super().__init__！
class MyList(list):
    def __init__(self, *args, **kwargs):
        super().__init__(*args, **kwargs)   # 让 list 完成初始化
        # 额外的初始化代码
        self.history = []
```

### 2.2 super() 的参数（通常不需要手动提供）

```python
# Python 3 中 super() 无参数形式（由编译器自动填充）：
super()
# 等价于（Python 2 风格，一般不需要这样写）：
super(CurrentClass, self)

# super(type, obj_or_type) 的两个参数：
# type：从哪里开始在 MRO 中搜索（默认：super() 所在的类）
# obj_or_type：接收者（默认：self）

# 什么时候需要显式提供参数？
# - 调试时跳过部分 MRO
# - 需要绕开特定超类的行为
# （普通情况下永远不需要）
```

### 2.3 super() 支持协作多重继承

```python
class Root:
    def ping(self):
        print(f'{self!r}.ping() in Root')


class A(Root):
    def ping(self):
        print(f'{self!r}.ping() in A')
        super().ping()          # ← 协作调用：继续往上找


class B(Root):
    def ping(self):
        print(f'{self!r}.ping() in B')
        super().ping()          # ← 协作调用


class Leaf(A, B):
    def ping(self):
        print(f'{self!r}.ping() in Leaf')
        super().ping()          # ← 协作调用


leaf = Leaf()
leaf.ping()
# 输出：
# <Leaf>.ping() in Leaf
# <Leaf>.ping() in A         ← super() 按 MRO 继续到 A
# <Leaf>.ping() in B         ← A.super() 按 MRO 继续到 B（不是 Root！）
# <Leaf>.ping() in Root      ← B.super() 按 MRO 继续到 Root

print(Leaf.__mro__)
# (<class 'Leaf'>, <class 'A'>, <class 'B'>, <class 'Root'>, <class 'object'>)
# super() 在 A 中找到的不是 Root，而是 MRO 中 A 的下一个：B
```

## 3. 子类化内置类型很麻烦

### 3.1 问题：C 语言实现的内置类型忽略用户覆盖的方法

```python
# 示例：dict 的 __init__ 和 update 会忽略用户覆盖的 __setitem__

class DoppelDict(dict):
    """每个值存两次（演示用）。"""

    def __setitem__(self, key, value):
        super().__setitem__(key, [value] * 2)    # 值重复两次


dd = DoppelDict(one=1)
print(dd)           # {'one': 1}          ← 问题！__init__ 没调用 __setitem__

dd['two'] = 2
print(dd)           # {'one': 1, 'two': [2, 2]}  ← [] 调用了覆盖的 __setitem__

dd.update(three=3)
print(dd)           # {'three': 3, ...}   ← 问题！update 也没调用 __setitem__

# 结论：
# - dict 的 __init__、update、get 等方法使用 C 语言实现
# - 它们不调用 Python 层面覆盖的方法
# - 这违背了面向对象的"晚期绑定"原则
```

### 3.2 解决方案：继承 UserDict、UserList、UserString

```python
import collections


class DoppelDict2(collections.UserDict):    # ← 继承 UserDict，而非 dict
    def __setitem__(self, key, value):
        super().__setitem__(key, [value] * 2)


dd = DoppelDict2(one=1)
print(dd)           # {'one': [1, 1]}  ← 正确！__init__ 调用了 __setitem__

dd['two'] = 2
print(dd)           # {'two': [2, 2], 'one': [1, 1]}  ← 正确

dd.update(three=3)
print(dd)           # {'three': [3, 3], ...}  ← 正确！


# 为什么 UserDict 能正确工作？
# UserDict 是用 Python 实现的包装类
# 所有操作都委托给内部的 dict 实例，但通过 Python 方法实现
# 因此会正确调用用户覆盖的方法


# 同理：
class MyList(collections.UserList):        # 而非 list
    pass

class MyStr(collections.UserString):       # 而非 str
    pass
```

### 3.3 何时直接继承内置类型

```python
# 如果行为与内置类型差别很大，可以继承对应的 ABC
from collections.abc import MutableMapping

class MyMapping(MutableMapping):
    """完全自定义的映射类。"""

    def __init__(self):
        self._data = {}

    def __getitem__(self, key):
        return self._data[key]

    def __setitem__(self, key, value):
        self._data[key] = value

    def __delitem__(self, key):
        del self._data[key]

    def __iter__(self):
        return iter(self._data)

    def __len__(self):
        return len(self._data)

    # 继承 MutableMapping 免费获得：
    # keys(), values(), items(), get(), pop(), update(), setdefault() ...
```

## 4. 多重继承与方法解析顺序（MRO）

### 4.1 菱形问题（Diamond Problem）

```python
class Root:
    def ping(self):
        print(f'{self!r}.ping() in Root')

    def pong(self):
        print(f'{self!r}.pong() in Root')

    def __repr__(self):
        return f'<{type(self).__name__}>'


class A(Root):
    def ping(self):
        print(f'{self!r}.ping() in A')
        super().ping()       # ← 协作方法

    def pong(self):
        print(f'{self!r}.pong() in A')
        super().pong()       # ← 协作方法


class B(Root):
    def ping(self):
        print(f'{self!r}.ping() in B')
        super().ping()       # ← 协作方法

    def pong(self):          # ← 非协作方法！没有调用 super()
        print(f'{self!r}.pong() in B')


class Leaf(A, B):
    def ping(self):
        print(f'{self!r}.ping() in Leaf')
        super().ping()


leaf = Leaf()

# 调用 ping（所有方法都是协作的）
leaf.ping()
# Leaf.ping() in Leaf
# Leaf.ping() in A        ← MRO 顺序
# Leaf.ping() in B
# Leaf.ping() in Root

# 调用 pong（B.pong 不是协作的）
leaf.pong()
# Leaf.pong() in A        ← Leaf 没有 pong，从 MRO 中第一个找到的 A 开始
# Leaf.pong() in B        ← A.pong 调用 super()，到 B
# Root.pong 不会被调用！  ← 因为 B.pong 没有调用 super()
```

### 4.2 查看和理解 MRO

```python
# 查看 MRO 的方法
print(Leaf.__mro__)
# (<class 'Leaf'>, <class 'A'>, <class 'B'>, <class 'Root'>, <class 'object'>)

# 便利函数：打印 MRO
def print_mro(cls):
    print(', '.join(c.__name__ for c in cls.__mro__))

print_mro(Leaf)  # Leaf, A, B, Root, object

# Tkinter 的复杂 MRO 示例
import tkinter
print_mro(tkinter.Text)
# Text, Widget, BaseWidget, Misc, Pack, Place, Grid, XView, YView, object
```

### 4.3 MRO 计算规则（C3 算法）

```python
# C3 算法（不是简单的广度优先搜索）
# 规则：子类总是排在超类前面；多个超类按照声明顺序排列

class A: pass
class B: pass
class C(A, B): pass    # C 先是 A，再是 B

print(C.__mro__)
# (<class 'C'>, <class 'A'>, <class 'B'>, <class 'object'>)

# 声明顺序影响 MRO
class D(B, A): pass    # D 先是 B，再是 A

print(D.__mro__)
# (<class 'D'>, <class 'B'>, <class 'A'>, <class 'object'>)
```

### 4.4 super() 的动态行为

```python
# super() 不是静态的，它依赖运行时的 MRO

from diamond import A  # A 来自之前的例子

class U:
    def ping(self):
        print(f'{self!r}.ping() in U')
        super().ping()   # U 的 MRO 中 super() 后面是 object，没有 ping 会报错


# u = U()
# u.ping()  # AttributeError: 'super' object has no attribute 'ping'
# 因为 U.__mro__ = (U, object)，object 没有 ping

class LeafUA(U, A):     # ← 同时继承 U 和 A
    def ping(self):
        print(f'{self!r}.ping() in LeafUA')
        super().ping()

leaf2 = LeafUA()
leaf2.ping()
# <LeafUA>.ping() in LeafUA
# <LeafUA>.ping() in U         ← MRO 顺序：LeafUA, U, A, Root, object
# <LeafUA>.ping() in A         ← U.super() 此时找到的是 A（MRO 中 U 的下一个）
# <LeafUA>.ping() in Root      ← A.super() 找到 Root

print(LeafUA.__mro__)
# (LeafUA, U, A, Root, object)
# 此时 U.super() 在 MRO 中找到 A，而不是 object！这就是"动态"的体现
```

## 5. 混入类（Mixin）

### 5.1 混入类的特征和规则

```
混入类的特征：
✅ 提供方法实现，供其他无关类重用
✅ 名称以 Mixin 结尾（约定）
✅ 每个方法都调用 super()（协作）
❌ 不能单独实例化
❌ 不保存内部状态（无实例属性）
❌ 不体现"是什么"关系（不代表一种类型）

使用混入类时：
- 在继承列表中，混入类必须排在第一位（比具体基类更靠前）
- 这确保混入类的方法在 MRO 中优先于具体基类的方法
```

### 5.2 实用混入类示例：不区分大小写的映射

```python
import collections


def _upper(key):
    """把字符串键转换为大写，非字符串键原样返回。"""
    try:
        return key.upper()
    except AttributeError:
        return key


class UpperCaseMixin:
    """混入类：让映射的键不区分大小写。

    混入的 4 个映射基本操作方法，每个都调用 super()，
    把转换为大写的键传给下一个实现。
    """

    def __setitem__(self, key, item):
        super().__setitem__(_upper(key), item)

    def __getitem__(self, key):
        return super().__getitem__(_upper(key))

    def get(self, key, default=None):
        return super().get(_upper(key), default)

    def __contains__(self, key):
        return super().__contains__(_upper(key))


# 混入必须排在第一位！
class UpperDict(UpperCaseMixin, collections.UserDict):
    pass
    # UpperCaseMixin 的方法优先处理键（转大写）
    # 然后通过 super() 委托给 UserDict 的实现

class UpperCounter(UpperCaseMixin, collections.Counter):
    """大小写不敏感的计数器。"""
    # UpperCaseMixin 与 Counter 结合：先转大写，再计数


# 测试
d = UpperDict([('a', 'letter A'), (2, 'digit two')])
print(list(d.keys()))       # ['A', 2]    ← 字符串键自动大写
d['b'] = 'letter B'
print('b' in d)             # True        ← 大小写不敏感的查找
print(d['a'], d.get('B'))   # 'letter A', 'letter B'

c = UpperCounter('BaNanA')
print(c.most_common())      # [('A', 3), ('N', 2), ('B', 1)]  ← 大写计数
```

### 5.3 混入类的工作原理（MRO 视角）

```python
# UpperDict 的 MRO：
print(UpperDict.__mro__)
# (UpperDict, UpperCaseMixin, UserDict, MutableMapping, ..., object)

# 当执行 d['a'] = 'letter A' 时：
# 1. Python 查找 __setitem__，在 MRO 中找到 UpperCaseMixin.__setitem__
# 2. UpperCaseMixin.__setitem__ 把 key 转为大写（'a' → 'A'）
# 3. 调用 super().__setitem__('A', 'letter A')
# 4. super() 在 MRO 中找到 UserDict.__setitem__
# 5. UserDict.__setitem__ 实际存储键值对

# 如果混入类不是第一个：
class WrongOrder(collections.UserDict, UpperCaseMixin):
    pass
# UpperDict 的 MRO：(WrongOrder, UserDict, UpperCaseMixin, ...)
# UserDict.__setitem__ 先被调用，跳过了 UpperCaseMixin！
# 结果：大小写转换不会发生
```

## 6. 多重继承的实际运用

### 6.1 `collections.abc` 中的抽象基类也是混入

```python
# ABC 中的具体方法叫"混入方法"
from collections.abc import MutableMapping

# MutableMapping 中的混入方法（无须实现，继承得到）：
# keys(), values(), items(), get(), pop(), setdefault(), update()

class MyMapping(MutableMapping):
    # 只需实现 5 个抽象方法：
    def __getitem__(self, key): ...
    def __setitem__(self, key, value): ...
    def __delitem__(self, key): ...
    def __iter__(self): ...
    def __len__(self): ...
    # 其余方法自动继承！
```

### 6.2 ThreadingMixIn：标准库中的混入类

```python
# http.server 模块的源码示意（Python 3.10）

import socketserver


class ThreadingMixIn:
    """混入类：在新线程中处理每个请求。"""

    def process_request_thread(self, request, client_address):
        """在单独的线程中处理请求（没有调用 super()，因为是新方法）。"""
        try:
            self.finish_request(request, client_address)
        except Exception:
            self.handle_error(request, client_address)
        finally:
            self.shutdown_request(request)

    def process_request(self, request, client_address):
        """覆盖 BaseServer.process_request，在新线程中处理请求。"""
        # 没有调用 super()，完全替换了 BaseServer 的同步实现
        t = threading.Thread(
            target=self.process_request_thread,
            args=(request, client_address)
        )
        t.start()

    def server_close(self):
        super().server_close()      # ← 协作：调用超类的 server_close
        self._threads.join()        # ← 等待所有线程结束


# 使用方式：
class ThreadingHTTPServer(socketserver.ThreadingMixIn, HTTPServer):
    daemon_threads = True
# 一行代码，HTTPServer 变成多线程服务器！
```

### 6.3 Django 泛化视图混入类

```python
# Django 的基于类的视图使用多重继承的最佳实践
# 每个混入类职责单一，名称以 Mixin 结尾

from django.views import View               # 基类：提供 dispatch 机制
from django.views.generic.base import (
    TemplateResponseMixin,                  # 混入：提供模板渲染能力

# 使用方式（Django 开发者熟悉的方式）：
class BookListView(ListView):
    model = Book
    template_name = 'books/list.html'
```

## 7. 应对多重继承的7条建议

### 建议1：优先使用对象组合，而不是类继承 ⭐

```python
# ❌ 过度继承：Widget 从三个几何管理器继承，但"小组件"不是"几何管理器"
class Widget(BaseWidget, Pack, Place, Grid):
    pass

# ✅ 更好：用组合，Widget 包含一个几何管理器引用
class Widget:
    def __init__(self, geometry_manager=None):
        self._geometry = geometry_manager or Pack()

    def pack(self, **kwargs):
        self._geometry.pack(**kwargs)   # 委托给几何管理器

# 组合和委托可以取代混入，但不能取代接口继承（定义类型层次）
```

### 建议2：理解不同情况下使用继承的原因

```python
# 使用继承的两种情况：

# ① 继承接口：建立"是什么"关系
class Dog(Animal):      # Dog 是一种 Animal
    pass

# ② 继承实现：重用代码
class CachingDict(MutableMapping):  # 重用 MutableMapping 的混入方法
    pass

# 最好能明确区分这两种情况：
# - 接口继承 → 首选 ABC 作为基类
# - 实现继承 → 考虑是否可以用混入或组合
```

### 建议3：使用抽象基类显式表示接口

```python
import abc


# ✅ 用 ABC 定义接口（显式）
class ISerializer(abc.ABC):
    @abc.abstractmethod
    def serialize(self, data: dict) -> str: ...

    @abc.abstractmethod
    def deserialize(self, data: str) -> dict: ...

class JsonSerializer(ISerializer):
    def serialize(self, data): return json.dumps(data)
    def deserialize(self, data): return json.loads(data)

# 或者用 typing.Protocol（现代 Python 推荐）
from typing import Protocol

class SerializerProtocol(Protocol):
    def serialize(self, data: dict) -> str: ...
    def deserialize(self, data: str) -> dict: ...
```

### 建议4：通过混入明确代码重用

```python
# 混入的命名约定：名称以 Mixin 结尾
class JsonMixin:
    """提供 JSON 序列化/反序列化的混入。"""

    def to_json(self) -> str:
        import json
        return json.dumps(self.__dict__)

    @classmethod
    def from_json(cls, json_str: str):
        import json
        data = json.loads(json_str)
        obj = cls.__new__(cls)
        obj.__dict__.update(data)
        return obj


class LogMixin:
    """提供日志功能的混入。"""

    def log(self, message: str):
        print(f'[{type(self).__name__}] {message}')


# 使用混入：混入在前，具体基类在后
class JsonLoggedModel(JsonMixin, LogMixin, BaseModel):
    pass
```

### 建议5：为用户提供聚合类

```python
# 聚合类：主体通常为空，只是把有用的类组合在一起

# Django 的 ListView 就是聚合类
class ListView(MultipleObjectTemplateResponseMixin, BaseListView):
    """渲染对象列表的视图。"""
    # 主体为空，但提供了所有功能的组合

# 自定义聚合类示例：
class RestfulMixin(JsonMixin, AuthMixin, CacheMixin):
    """RESTful API 视图的聚合类。"""
    # 使用者只需继承这一个类，无须记住哪些混入要一起使用
```

### 建议6：仅子类化为子类化设计的类

```python
from typing import final


# 使用 @final 标记不应被子类化的类（Python 3.8+）
@final
class Token:
    """安全令牌类：禁止子类化。"""
    def __init__(self, value: str):
        self._value = value


# 使用 @final 标记不应被覆盖的方法
class BaseProcessor:
    @final
    def process(self, data):
        """模板方法：子类不应覆盖这个方法。"""
        validated = self.validate(data)     # 调用钩子方法
        return self.transform(validated)

    def validate(self, data):               # 钩子方法：子类可以覆盖
        return data

    def transform(self, data):              # 钩子方法：子类可以覆盖
        return data
```

### 建议7：避免子类化具体类

```python
# ❌ 危险：子类化具体类（可能破坏内部状态）
class ExtendedList(list):
    def append(self, item):
        # 这里修改了行为，但 list 的其他方法（extend、__init__ 等）
        # 可能不会调用这个覆盖的 append！
        super().append(item * 2)

bad = ExtendedList()
bad.append(1)          # [2]   ← 正确（直接调用覆盖的 append）
bad.extend([2, 3])     # [2, 2, 3]  ← 问题！extend 不调用覆盖的 append
bad += [4, 5]          # [2, 2, 3, 4, 5]  ← 同样的问题


# ✅ 更好：继承 UserList（Python 实现，正确调用覆盖方法）
class ExtendedList2(collections.UserList):
    def append(self, item):
        super().append(item * 2)

good = ExtendedList2()
good.append(1)         # [2]
good.extend([2, 3])    # [2, 4, 6]  ← 正确！extend 调用了覆盖的 append

# 或者：继承抽象基类，从头实现
# 或者：用组合替代继承
```

## 8. 本章知识速查表

### 继承场景选型

| 场景 | 推荐方案 | 原因 |
|------|---------|------|
| 定义接口/类型层次 | 继承 abc.ABC | 明确表达，支持抽象方法检查 |
| 扩展 dict/list/str | 继承 UserDict/UserList/UserString | 避免 C 层方法忽略覆盖的问题 |
| 完全自定义集合 | 继承 collections.abc 中的 ABC | 只需实现核心方法，其余免费获得 |
| 多种类型共享行为 | 定义混入类（Mixin）| 清晰表达意图，不构成类型关系 |
| 想重用代码但不继承 | 组合/委托 | 更灵活，低耦合 |

### MRO 关键规则

```
MRO 的规律（C3 算法的简化理解）：
1. 子类总是排在超类之前
2. 多个超类按照 class 声明时的顺序排列
3. 从最左边开始，深度优先，但要"合并"菱形路径

Leaf(A, B) 的 MRO：Leaf, A, B, Root, object
# 不是 Leaf, A, Root, B（深度优先）
# 而是保证 B 在 Root 之前（如果 B 是 Root 的子类）
```

### 混入类使用规范

| 规范 | 说明 |
|------|------|
| 命名 | 类名以 `Mixin` 结尾 |
| 位置 | 在继承列表中排在最前面 |
| 方法 | 每个方法都调用 `super()`（协作方法）|
| 状态 | 不保存实例属性（无内部状态）|
| 实例化 | 不单独实例化 |
| 关系 | 不代表"是什么"，只提供行为 |

### super() vs 直接调用超类

| | `super().__method__()` | `SuperClass.__method__(self)` |
|--|---|---|
| **多重继承支持** | ✅ 完全支持（遵循 MRO）| ❌ 跳过其他超类的同名方法 |
| **代码耦合** | 松耦合（不硬编码基类名）| 紧耦合（硬编码了基类名）|
| **协作调用链** | ✅ 支持 | ❌ 不支持 |
| **推荐程度** | ✅ 推荐 | ❌ 不推荐（特殊情况除外）|
